༣༢༣༽ ༄༅། བླ་མ་དགོངས་པ་འདུས་པའི་སྒྲུབ་པ་ཆེན་པོ་བདུད་རྩི་སྨན་སྒྲུབ་སྤྲོས་བཅས་བུམ་དབང་དང་བཅས་པ་ཕྱག་བཞེས་ཇི་ལྟར་བྱ་ཚུལ་གྱི་ས་བཅད་ཟིན་དུ་བཀོད་པ་བཞུགས་སོ། ༢༣༤༽ གདོད་ནས་རང་བཞིན་རྣམ་དག་ཆོས་སྐུའི་ཀློང༌། །འགག་མེད་རང་ཤར་གཟུགས་སྐུ་དཀྱིལ་འཁོར་ལྷ། །སྣ་ཚོགས་སྤྲུལ་བསྒྱུར་སྒྱུ་མའི་གར་མཁན་མཆོག་ བླ་མ་ལྷ་དང་ཌཱ་ཀིར་སྙིང་ནས་འདུད། །ཐེག་དགུའི་ཡང་ཞུན་ཟབ་གསལ་རྡོ་རྗེའི་ཐེག་ །མཐའ་དག་སྨིན་གྲོལ་གནད་གསང་ལ་འདུས་ཕྱིར། །ཟབ་པའི་མཐར་ཐུག་དགོངས་འདུས་གསེར་ཆོས་ཀྱི། །སྨིན་གྲོལ་ཕྱག་བཞེས་རིམ་པ་འདིར་བསྟན་བྱ། །དེ་ལ་འདིར་བླ་མ་དགོངས་པ་འདུས་པའི་སྨིན་བྱེད་ཀྱི་སྒྲུབ་པ་ཆེན་པོའི་ཕྱག་བཞེས་རིག་འཛིན་བརྒྱུད་པའི་ ༣༢༥༽ ཞལ་གྱི་བདུད་རྩི་དྲི་མ་མེད་པ་ཡི་གེར་བཀོད་དེ་སྐལ་ལྡན་རྣམས་ཀྱི་འདོད་པ་འཇོ་བའི་ནོར་བུར་སྦྱིན་པར་བྱ་བ་ལ། ཐོག་མར་ས་བཅད་ཀྱི་བསྡོམས་ནི། སྙན་གསན་དབབ་ཅིང་སྒྲུབ་རྫས་ཡོ་བྱད་བསག་ ས་ཡི་བདག་པོ་རྣམས་ལ་བདེན་ཚིག་བརྗོད། ཆུ་གཏོར་སཱཙྪ་རྒྱ་བཞི་གསུངས་རབ་བསྒྲག་ གཏོར་ཟློག་ལིང་སྒྲུབ་དྲག་པོའི་སྦྱིན་སྲེག་བྱ། དེ་ལྟར་རྐྱེན་ཟློག་ཚོགས་གསོག་ཅི་འགྲུབ་འབད། དེ་ནས་སྔོན་འགྲོར་སྐོངས་དང་ཡིག་བརྒྱ་འབུལ། བརྒྱ་ཕྱག་སྤང་སྐོང་ཞི་བའི་སྦྱིན་ ༣༢༦༽ སྲེག་བྱ། །རྟགས་ཀྱི་ཕྱག་རྒྱས་རྣལ་འབྱོར་གྲངས་ལ་གདབ། །ས་དབྱིབས་སྤྱི་དང་བྱེ་བྲག་བརྟག་ལ་བླངས། །སྔགས་དང་བྱ་བས་སྦྱང་ཞིང་མ་ཏྲཾ་བསྒྲལ། །གཏེར་སྦ་བློ་དང་སྟངས་སྟབས་ཁྲོ་གར་བཟུང༌། །ཕུར་བུས་བསྲུང་ཞིང་ཤིས་བརྗོད་གར་སྡུད་བྱ། །ཚོམ་བུའི་དཀྱིལ་འཁོར་བཞེངས་ཤིང་བུམ་སོགས་བཤམས། །སྐྱབས་སེམས་ཡན་ལག་བདུན་དང་ཕྱི་གཏོར་བཏང༌། །བདག་སྐྱེད་རྒྱལ་ཐོ་གཟུག་ཅིང་གསོལ་བ་གདབ། །མདུན་སྐྱེད་བུམ་སྐྱེད་སོ་ཤིང་སོགས་བྱིན་བརླབ། ཆོས་སྐྱོང་བསྐང་གསོ་ཁྲུས་དང་མེ་ཐུན་བརྡེག་ །དམ་སྲི་མནན་ཞིང་སྒོ་བྱང་ཕྱི་ནང་གདག་

།ཀུན་ཚོགས་བྷུར་ཕུད་བགེགས་སྦྱང་ཌཱཀྐི་འགྱེད། །མཚམས་ཕུར་ཕྱག་འཚལ་མཎྜལ་ཆོས་བཤད་བྱ། །སྐྱབས་སེམས་སྡོམ་བཟུང་གནས་གསུམ་བྱིན་གྱིས་བརླབ། །སོ་ཤིང་ཁྱོར་ཆུ་སྲུང་སྐུད་ཀུ་ཤ་སྦྱིན། །ལོ་རྒྱུས་བཤད་ནས་ཤིས་བརྗོད་འདུ་ཚོགས་དགྱེས། །བཏང་རག་ཚོགས་མཆོད་གཏོར་བསྔོ་གསོལ་གདབ་བྱ། །ལྷ་ཚོགས་ནམ་མཁར་བཏེགས་ནས་རྫས་རྣམས་བསྡུའོ། སྔོན་འགྲོ་ཐུན་མོང་ཁྱད་པར་ཡོངས་རྫོགས་པའོ། ༈ དངོས་གཞི་དང་པོར་ཆག་ཆག་ཐིག་ཚོན་བྲི། །བུམ་སོགས་བཅས་བཤད་ལྷ་ཚོགས་བསྐོས་ལ་གདབ། །དཀྱིལ་འཁོར་རྒྱན་དང་སྒྲུབ་རྫས་ཤར་གདབ་དགོད། །བསངས་སྦྱངས་སྐྱབས་སེམས་ཚོགས་བསགས་སྲུང་མཚམས་བཅད། །ཆོས་བཤད་སློབ་དཔོན་མཆེད་ལྕམ་མངའ་གསོལ་འབུལ། །རྟགས་གདབ་བརྡའ་སྒོ་བརྡའ་ཕྱག་བྱིན་འབེབས་བྱ། །ཡན་ལག་བདུན་པ་བདག་མདུན་བུམ་སྐྱེད་སོགས། །དཀྱིལ་ཆོག་བསྐང་གསོ་ཚོགས་ལས་ཡོངས་རྫོགས་བཏགས། །ནང་པར་དཀྱིལ་ཆོག་ཡོངས་རྫོགས་གཞུང་བསང་ལ། །དཀྱིལ་འཁོར་ཕྱི་འཇུག་ནང་འཇུག་ལྷ་སྔགས་ཀྱི། །དག་པ་སོ་སོར་རྒྱས་བཤད་འཛབ་དགོངས་ཕོགས། །འཛབ་བསྐུལ་ཞག་གསུམ་ལ་སོགས་ ༣༢༨༽ ཅི་རིགས་སྒྲུབས། ། དེ་ཚེ་སྨན་དང་སྦྲེལ་ན་ཁྲོམ་བུ་བཤམ། །སྦྱང་དང་བཞུ་བཏུལ་འཛབ་བཅས་བསྒྲུབ་བྱས་ལ། །སླར་ཡང་བཏགས་བཙགས་སྦོལ་གཞལ་རྒྱས་བཏབ་བཤམས། །རྟ་བྲོས་བགེགས་བསྐྲད་སྨན་གྱི་བྱིན་རླབས་ནི། །དུས་དྲུག་མྱུལ་ལེ་བསྐྲད་བཅས་འཛབ་འབུངས་སྒྲུབས། །རྟ་ཐེན་ནུ་མོ་ལས་བྱང་བསྟོད་བསྐུལ་འབད། །མཎྜལ་ཞལ་བསྒྱུར་ཚེ་དབང་རྫས་སོགས་བཤམས། །ཐོ་རངས་སྒྲུབ་ཁོག་གཞུང་ལྟར་དངོས་གྲུབ་བླངས། །བདུད་རྩིའི་དངོས་གྲུབ་ལེན་ཆོག་སྨན་དབང་བསྐུར། །རྒྱ་དཀྲོལ་སྨན་མཆོད་དངོས་གྲུབ་རྫས་ཆོག་རོལ། །ཚེ་དབང་བསྐུར་ཞིང་ཕྱི་དྲོ་སྒྲུབ་ལུང་སྦྱིན། རྒྱས་པའི་སྦྱིན་སྲེག་གཏུམ་མོའི་མེ་དབང་བྱ། ། ༈ དེ་ནས་སྤྲོས་བཅས་བུམ་དབང་མཐའ་རྟེན་བཅས། །

རབ་འབྱམས་སྐུ་གསུང་ཐུགས་ཡོན་ཕྲིན་ལས་དང༌། །གསང་བ་རྡོ་རྗེ་སློབ་དཔོན་དོན་དམ་བཅུད། །ཡོན་ཏན་མཆོད་རྫས་ཕྲིན་ལས་སྲོག་དབང་དང༌། །དབང་གི་བྱིན་ ༣༢༩༽ སྲེག་ལ་སོགས་སྐབས་བསྐུར་བྱ། ། དངོས་གཞི་སྒྲུབ་སྨན་དབང་ཀུན་ཡོངས་རྫོགས་པའོ། །རྗེས་ནི་དྲག་ལས་བཅས་གསར་བསྐུལ་ཆེན་བཏགས། ། མཁྱེན་གསོལ་གཟུགས་དཔང་བསྒྲལ་ལས་ཆོས་སྐྱོང་བསྐུལ། །ཕྱི་ནང་ཐོད་དཀྲོལ་བཏང་རག་ཟོར་སྒོ་དབྱེ། ། འཆད་མོད་གཏོར་ཟོར་བསྟན་སྐྱོང་གཏོར་བྲོ་བྱ། །བཏང་རག་མཆོད་བསྟོད་དངོས་གྲུབ་གཏོར་དབང་ཞུ། །གཤེགས་གསོལ་བསྔོ་སྨོན་བསྡུ་མཆོག་ཀུན་གྲུབ་ཚེ། །རྡོ་རྗེ་སློབ་དཔོན་སྟེགས་བུའི་སྟེང་བཞུགས་ལ། །སྐུ་དྲིན་གསོ་ཞིང་མར་མེ་སྨོན་ལམ་གདབ། །ཤིས་བརྗོད་གཏེར་སྦ་རྡུལ་ཚོན་གཤེགས་གནས་བསྲུང༌། །རྗེས་ཀྱི་རིམ་པ་མཐའ་དག་ཡོངས་གྲུབ་པའོ། ། ༈ དེ་ལྟ་བུས་སྒྲུབ་ཆེན་གྱི་ས་བཅད་སྤྱི་སྡོམ་གྱི་རིམ་པ་མདོར་བསྟན་ནས། བྱེ་བྲག་ཏུ་ཕྱག་བཞེས་ཀྱི་ལས་རིམ་ཇི་ལྟར་བྱ་ཚུལ་རྒྱས་པར་བཤད་པ་ནི། དེ་ཡང་གསང་སྔགས་ཐེག་ཆེན་བླ་ ༣༣༠༽ མེད་ཀྱི་དབང་དང་སྒྲུབ་པ་ཆེན་པོ་འདི་ལྟ་བུ་སྟོན་པར་བྱེད་པའི་རྡོ་རྗེའི་སློབ་དཔོན་མཚན་ཉིད་དང་ལྡན་པ་གང་དེ་ཉིད། སྔོན་དུ་བདག་ཉིད་རྩ་བ་གསུམ་གྱི་བསྙེན་པ་གྲངས་དུས་རྟགས་ཀྱི་རིམ་པ་དང་བཅས་པ་གཞུང་དང་མཐུན་པར་ཚད་དུ་འཁྱོལ་བ་དང༌། རང་དོན་དུ་སྤྲོས་མེད་ཀྱི་སྒྲུབ་པའི་རིགས་གང་དང་གང་དག་རྣམས་་ཀྱང་ནུས་ལྡན་རྟགས་ཐོན་དུ་སྒྲུབས་པ་ནི་གཅེས་པ་ཁོ་ན་ཡིན་ཕྱིར་དེ་ལྟ་བུ་སྔཽན་དུ་བཏགས་ལ། རིགས་གང་དང་གང་རྣམས་ཀྱང་ནུས་ལྡན་རྟགས་ཐོན་དུ་སྒྲུབས་པ་ནི་མཆོག་ཏུ་གཅེས་པ་ཁོ་ན་ཡིན་ཕྱིར་དེ་ལྟ་བུ་སྔོན་དུ་བཏང་ལ། ༈ དེ་ནས་མཚན་ཉིད་དང་ལྡན་པའི་སློབ་མ་རྣམས་ཀྱིས་ཀྱང་ལོ་ཟླའོ་གོང་ནས་སློབ་དཔོན་ལ་སྙན་གསན་དབབ་པ་དང༌། བླ་མས་ཀྱང་སྒྲུབ་པ་ཆེན་པོ་ལ་དགོས་པའོ་ཡོ་བྱད་ཀྱི་སྟ་གོན།

བསྙེན་པ་ཁ་གསོ་བའི་རིགས་སོགས་གྲུབ་པར་བྱས་ལ། དེ་ནས་སྒྲུབ་པ་དངོས་ཀྱི་དུས་ཚེས་ཕུན་སུམ་ཚོགས་པའི་སྐབས་ལ་ནམ་བབས་པའི་ཚེ། ཐོག་མར་ཐུན་མོང་གི་སྔོན་དུ་འགྲོ་བའི་ལས་ཁྲིགས་ལ། དང་པོ་ས་ཆོག་སྟ་གོན་ ༣༣༡༽ ལས། ས་བསླང་ནི། ཡུལ་ཅན་མི་ལ་བསླང་བ་ནི་བཟའ་བཏུང་བདེན་ཚིག་དཀྲོལ་བ་སོགས་ཀྱིས་བསླང་ཞིང༌། མི་མ་ཡིན་རྣམས་ལ་ནི་བསངས་དང་བསུར་མཆོད་ལ་སོགས་པས་ཚིམ་པར་བྱས་ཏེ་ས་བསླང་བའི་ཆོ་ག་བྱ་ཞིང་མ་མོའི་མདོས་སམ། རྒྱལ་མདོས་དང་ལྷ་འདྲེ་སྟོང་ཕྲག་སོགས་ཀྱི་བར་ཆད་བསལ། རི་རྒྱ་ཀླུང་རྒྱ་ལམ་རྒྱ་དམག་རྒྱ༌། རྒྱ་བཞི་བསྡམས་པ། སཱཙྪ། ཆུ་གཏོར། གསུངས་རབ་བསྒྲགས་པ་སོགས་ཀྱི་རྒྱབ་ཆོས་དཀར་ཆོས་གང་འགྲུབ་དང༌། བཀའ་བརྒྱད་དང་ཕུར་པ་གང་རུང་གི་གཏོར་སྒྲུབ་ལོངས་སྐབས་བསྟུན་པའི་ཞག་གྲངས་ཀྱི་སྒྲུབས་ནས། དམ་སྲི་མནན་པའི་ལིངྒ་དང་ཐུན་ལ་སོགས་ཀྱང་འདིའི་ཁོངས་ནས་ནན་ཏན་དུ་བསྒྲུབ་ཅིང་གཏོར་ཟློག་ཚུལ་ལྡན་དུ་བྱ་བ་དང༌། དྲག་པོའི་སྦྱིན་སྲེག་ལ་དགོས་པའི་ལིངྒ་ཡང་བརྒྱ་ཕྲག་ཙམ་དུ་སྟ་གོན་བྱས་ལ་འདི་དུས་ལྷན་ཅིག་ཏུ་བསྒྲུབ་པའི་ ༣༣༢༽ མཐར། སྒྲུབ་པ་ཆེན་པོ་ལ་བར་དུ་གཅོད་པའི་བགེགས་རིགས་རྡུལ་དུ་རློག་པའི་དྲག་པོའི་སྲེག་ཀྱང་རྒྱས་པར་བྱ་བ་རབ་ཏུ་གཅེས་ཏེ། མདོར་ན་འདི་ཡན་ཆད་མཐའ་དག་ནི་སྒྲུབ་པའི་སྔོན་ནས་བར་ཆད་བསལ་བའི་རྟེན་འབྲེལ་གྱི་ཆོ་ག་ཡིན་ཕྱིར། སྒྲུབ་པའི་ཞག་གྲངས་སྒོང་ལ་མི་ཐེ་བ་ཉིད་དོ། ༈ དེ་ནས་སྔོན་འགྲོ་དངོས་ཀྱི་ཞག་དང་པོ་ལ། ན་རག་སྐོངས་བཤགས་ཡིག་བརྒྱ་སྤང་སྐོང་ཕྱག་རྒྱ་ལ་སོགས་སྒྲུབ་པ་པོ་ཐམས་ཅད་ཀྱི་ཉམས་ཆགས་བཤགས་པའི་ཐབས་ལ་འབད་ཅིང༌། སྡིག་སྒྲིབ་སྦྱོང་ཕྱིར་ལ་དམིགས་ནས་ཞི་བའི་སྦྱིན་སྲེག་དང་དེའི་བླུད་ཆོག་སོགས་གནད་ཀྱི་ཞལ་གདམས་ནས་འབྱུང་བ་ལྟར་བྱ་བ་ཡང་གཏེར་སྟོན་བརྒྱུད་པའི་ཕྱག་བཞེས་ཡིན་ལ། འདི་ནས་བཟུང་སྟེ་སྒྲུབ་ཆེན་ལ་ཞུགས་མཁན་ཇི་ཙམ་ཡོད་པའི་གྲངས་དང་མཐུན་པའི་རྡོ་རྗེ་སློབ་དཔོན་གྱི་ཕྱག་རྟགས་བཀྲམ་ནས་གྲངས་ངེས་སུ་བྱས་ལ།

ཕྱག་རྟགས་དང་མི་ལྡན་པ་ནི་སྒྲུབ་པའི་ཁོངས་སུ་མི་གཞུག་པ་ཡང་བར་ཆད་མི་འབྱུང་བའི་རྟེན་འབྲེལ་ཐབས་གནད་ཟབ་མོ་སྟེ། དེ་ལྟར་འདི་ཡན་ཆད་སྔོན་འགྲོའི་ཞག་དང་པོའོ། ཞག་གཉིས་པ་ལ། ས་བརྟག་ བླང་བ། སྦྱང་བ་རྣམས་དཀྱིལ་ཆོག་ཆེན་པོ་ལས་འབྱུང་བ་བཞིན་དང༌། སྦྱང་བ་ཚུལ་བཞིན་བྱས་པའི་ས་དེ་ཉིད་སྟེགས་བུའི་གནས་སུ་སྤུངས་པ་ལ་མ་ཏྲཾ་གཟུགས་གཞུང་བཞིན་བཟོས་ལ། གིང་དང་རོལ་མོ་བ་སོགས་ཚགས་སུ་ཆུད་པ་དང༌། ཐམས་ཅད་ཚོགས་ཏེ་བླ་མས་འདུས་པ་རྣམས་ལ་མ་ཏྲཾ་གྱི་ལོ་རྒྱུས་བཤད་ལ། མཁྱེན་གསོལ་གཟུགས་དཔང་མ་ཏྲཾ་བསྐྱེད་ཆོག་དགུགས་བསྟིམ་དྲག་པོ་ལན་མང་པོ་དང་བཅས་པ་བྱ་ཞིང༌། འདི་སྐབས་ཀྱི་བདག་བསྐྱེད་ནི་མངོན་རྟོགས་ཆེན་མོ་ལྟར་གྱི་རྟ་མགྲིན་ཁོ་ན་བསྐྱེད་པར་བསྔགས་སོ། དེ་ནས་གིང་པ་རྣམས་འཐེན་པས་ ༣༣༤༽ སྒྲུབ་ཁོག་གི་གཞུང་ནས་ཇི་ལྟར་གསུངས་པའི་མ་ཏྲཾ་མནན་ཆོག་ཚུལ་བཞིན་བྱས་ནས། རྡོ་རྗེ་སློན་དཔོན་ཉིད་དེའི་དབུས་སུ་གར་དབྱངས་དང་བཅས་པས་མཏྲཾ་མནན་ཆོག་བྱ། སྟེགས་བུ་རྒྱུན་ཞག་ཏུ་བཟོ་ན་འདི་སྐབས་ས་གཏཻར་སྦ་བ་སྤྱི་དང་མཚུངས་པར་བྱ་བ་ཡིན་ལ། དེ་ནས་ས་བཟུང་བ་ལ་བློས་བཟུང་ཚུལ་ནི་གཞུང་ཉིད་ཀྱི་གོ་སླ་མོད་ཀྱི། སྟངས་སྟབས་ཀྱིས་བཟུང་བ་ནི། དཀྱིལ་འཁོར་གཙོ་བོ་རྡོ་རྗེ་སློབ་དཔོན་ནུས་ལྡན་གྱི་རྒྱལ་ཐོབ་བརྒྱད་དང་བཅས་པ་ཆས་ཀྱིས་རབ་ཏུ་བརྒྱན་ཞིང་རྡོ་རྗེ་དྲིལ་བུ་འཆང་བ་ཧཱུྃ་གི་གླུ་ཆེན་འབྱིན་བཞིན་པས་ལམ་ལ་ཞུགས་རྡོ་རྗེའི་འགྲོས་ཀྱིས་སྟེགས་བུ་བསྐོར་ནས། དབུས་དང་ཕྱོགས་མཚམས་སོ་སོར་འཁོད་པའི་མདུན་དུ་མཆོད་གཏོར་མདོར་བསྡུས་བཤམས་ཏེ། སྐྱབས་ ༣༣༥༽ སེམས་ཡན་ལག་བདུན་དང་ལྕེ་ལག་རྡོར་དྲིལ་མཆོད་གཏོར་བྱིན་རླབས་སོགས་ནས་བཟུང་སྟེ་གཙོ་བོ་དགུའི་བདག་བསྐྱེད་སྤྱན་འདྲེན་བཞུགས་གསོལ་དབང་བསྐུར་རྒྱས་གདབ་མཆོད་བསྟོད་བཟླས་པའི་བར་གཞུང་བསྲང༌། སློབ་དཔོན་གྱི་ལྷ་ལ་གསོལ་བ་གདབ་པ་དང༌།

ཕྱོགས་མཚམས་ཀྱི་རྒྱལ་ཐེབས་རྣམས་ཀྱིས་གཙོ་བོ་ཉིད་ལ་གསོལ་བ་བཏབ་པ་སོགས་གྲུབ་རྗེས་བཛྲ་ཨུ་ཏིཥྛ་ཡི་སྔགས་དང་ཕྱག་རྒྱས་ཡེ་ཤེས་བཏེགས་ལ། སླར་ལས་ཀྱི་རྟ་མགྲིན་བསྐྱེད་ནས། ཆོས་སྐུ་རྣམ་དག་མཚན་མ་ཀུན་སྤངས་ཀྱང༌། །ཞེས་སོགས་ཀྱི་བགེགས་རིགས་ལ་བཀའ་བསྒོ་ཞིང་བདེན་སྟོབས་བརྗོད་པས་བསྐྲད། དེ་ནས་ཁྲོ་གར་གྱི་བཟུང་བ་དངོས་ནི། འབྱུང་འདུས་དང་བསྐྲད་གཟིར་ལ་སོགས་པའི་སྟངས་སྟབས་ཀྱིས་ས་གཞིའི་བར་ཆད་བསལ། རྡོ་རྗེ་ཁྱུང་གཤོག་འགྲོས་དང་སྣ་ཚོགས་རྡོ་རྗེའི་འགྲོས་ལ་ ༣༣༦༽ སོགས་པས་ས་གཞི་ཟིལ་གྱིས་མནན་ཞིང་བཅགས། ཕར་ཕྱིན་ལྔའི་བགྲོས་ཀྱིས་ས་གཞི་བྱིན་གྱིས་བརླབ་པར་བྱ། དེ་རྗེས་ས་བསྲུང་ཀྱི་ངང་ནས་ལོག་པར་བྱ། དེ་རྗེས་ས་བསྲུང་བའི་ཕྱིར་བུ་ཁྲོ་བོར་མངའ་གསོལ་བ་དཀྱིལ་འཁོར་གྱི་སྟེགས་བུའི་བྱང་ཤར་མཚམས་སུ་བཏབ་ལ། གར་གྱི་སྡུད་འགྲོས་ཁམས་གསུམ་དབང་བསྡུད་ཀྱི་ངང་ནས་ལོག་པར་བྱའོ། ཞག་གསུམ་པ་ལ་དཀྱིལ་འཁོར་སྟ་གོན། བུམ་པ་སྟ་གོན། ལྷ་ལྟ་གོན། སློབ་མ་སྟ་གོན་དང་བཅས་པ་རྣམས་ལས། དང་པོ་ནི། སྟེགས་བུའི་ཁར་ཐིག་ཆེན་བརྒྱད་དང་ལིང་ཚེ་དགུ་ཐིག་བཏབ་ལ། ལྷ་སོ་སོའི་རྩ་སྔགས་བུམ་སྒྲུབ་དབང་ཆེན་མོ་ལྟར་དངབཅས་པས་དྲིའི་ཐིག་ལེ་བྱས་པའི་ཁར་ཚོམ་བུ་དགོད་དེ། དེ་ཡང་དབུས་ཀྱི་གཙོ་མཆོག་ལ་ཚོམ་བུ་གཞན་གྱི་ལྔ་འགྱུར་དང༌། འཁོར་གྱི་གཙོ་བརྒྱད་ལ་གསུམ་འགྱུར། འཁོར་ཕལ་བ་ལ་རེ་རཻ། སྒོ་མཚམས་དང་དཔའ་བོ་དཔའ་མོ་རྣམས་ལཚོམ་བུ་ཕལ་གྱི་སུམ་ ༣༣༧༽ ཆ་ཙམ་དུ་དགོད། བུམ་པ་སྟ་གོན་ནི། ཕྱག་བཞེས་ལ་ལར་དངོས་གཞིའི་རྒྱན་དགོད་ཀྱི་སྔོན་དེར་མཛད་པའང་ཡོད་མོད་ཀྱི། སྤྱི་འགྲོ་ལྟར་ན་འདིར་བྱ་བ་སྟེ། དེ་ཡང་རྡོ་རྗེ་སློབ་དཔོན་གྱི་མདུན་དུ་གདན་མཛེས་གཙང་མའི་ཁར་མེ་ཏོག་བཀྲམ་ལ། བུམ་པ་དགུ་ལས་བུམ་དང་བཅས་པ་རྣམས་གུ་གུལ་གྱིས་བདུག་ཅིང་བགེགས་བསྐྲད། ཁ་རྒྱན་མགུལ་འཆིང་གཟུངས་ཐག་བུམ་རྫས་ཆུ་སོགས་ཐམས་ཅད་བཤམས་ལ།

དཀྱིལ་ཆོག་ནས་འབྱུང་བ་ལྟར་གྱི་བུམ་པ་བཅའ་བའི་རིམ་པ་རྣམས་ཞིབ་པར་བྱ་སྟེ། བུམ་པ་རྣམས་དཀྱིལ་འཁོར་དབུས་སུ་ལྡིང་ཁྲིའི་ཁར་ཕྱོགས་མཚམས་མ་འཁྲུག་པར་དགོད། གཞན་ཡང་ལྷའི་སྐུ་རྟེན་ཙཀྐ་ལི་རྣམས་དང༌། ཕྱི་རིམ་དུ་མཚམས་ཕུར་སོ་ཤིང༌། སྲུང་སྐུད། ཁྱོར་ཆུ་ཀུ་ཤ་སོགས་སློབ་མ་སྟ་གོན་གྱི་ཡོ་བྱད་མཐའ་དག་ཚང་བར་བཤམས་པའི་དཀྱིལ་འཁོར་ ༣༣༨༽ གྱི་མཐའ་སྐོར་དུ་ཕྱི་ནང་གི་མཆོད་གཏོར་གྱིས་སྐོར་ལ། སྐྱབས་སེམས་ཡན་ལག་བདུན་སོགས་སྔོན་དུ་བཏང་སྟེ། སྔོན་གཏོར་ལ་བབས་པའི་ཚེ། དཔོན་སློབ་རྣམས་ཕྱི་རོལ་བླ་གབ་མེད་པར་ཕྱིན་ནས་བསངས་བསུར་གྱི་མཆོད་པ་དང་བཅས་པས་དཀར་ཕྱོགས་སྐྱོང་བའི་གཞི་བདག་ལ་གསེར་སྐྱེམས་ལན་ཆགས་བགེགས་རིགས་ཀྱི་འབྱུང་གཏོར་དང༌། སྤྱི་དང་མཐུན་པའི་ཆ་གསུམ་གཏོར་མ་སོགས་ཀྱང་རྒྱས་པར་བཤམས་ལ་དེ་དག་གི་ཆོ་གའི་རིམ་པ་རྣམས་ཀྱང་ཚུལ་བཞིན་དུ་བྱ། ནང་དུ་ལོག་ནས་བསྲུང་འཁོར་མཚམས་གཅོད་ནས་བཟུང་སྟེ་བདག་བདག་བསྐྱེད་འཛབ་ཡན་ཆད་གྲུབ་པ་དང༌། སླར་ཕྱི་རོལ་གྱི་ཕྱོགས་བཞིར་འཇིག་རྟེན་སྐྱོང་བའི་རྒྱལ་ཆེན་བཞི་པོའི་ཐོ་སྐུ་རྟེན་འཕན་དར་དང་བཅས་པ་གཙུག་ལ། དེ་དག་སོ་སོར་གསེར་སྐྱེམས་གཏོར་མཆོད་བཤམས་ནས་ཆོ་ག་གཉེར་གཏད་རབ་ ༣༣༩༽ གནས་ཤིས་བརྗོད་དང་བཅས་པ་གཞུང་ལྟར་ཕྱོགས་བཞིར་རིམ་ཅན་དུ་བྱས་ཏེ། ནང་དུ་ལོག་ཚེ་སློབ་དཔོན་གྱིས་དཀྱིལ་འཁོར་ས་ལ་གསོལ་གདབ་བྱ་ཞིང༌། སླར་རིམ་བཞིན་མདུན་བསྐྱེད་བུམ་བསྐྱེད་སོགས་དཀྱིལ་ཆོག་ཡོངས་རྫོགས་བཟླས་པ་དང་བཅས་པའི་མཐར། མཚམས་ཕུར་སོ་ཤིང་སོགས་ཐ་དད་ཀྱི་བསྐྱེད་བཟླས་དང༌། ཁྱོར་ཆུ་ཀུ་ཤ་སོགས་ཀྱི་བྱིན་རླབས་བཟླས་པ་སྔགས་ཐམས་ཅད་སྐབས་བསྟུན་ཅི་འགྲུབ་བྱ། ཆོས་སྐྱོང་རྣམས་ཀྱིས་བསྐང་གསོ། བཀའ་སྲུང་གཏེར་སྲུང་གི་གཏོར་མ་དགེར་འབུལ་སོགས་མདོར་ན་ཚོགས་ལས་ལ་ཐུག་གི་བར་མཐའ་དག་གྲུབ་པར་བྱ་ཞིང༌། དམ་སྲི་མནན་པའི་ལིང་སྒྲུབ་རྒྱས་པ་སྔོན་དུ་མ་སྒྲུབ་ན་ནི་འདི་དུས་གདབ་པ་ལས་དྲུག་ཐམས་ཅད་ཞིབ་པར་བྱ་དགོས་པ་ཉིད་དོ།

དེ་ནས་ཕྱི་རོལ་ཏུ་སློབ་མ་གྲངས་ངེས་གང་ཡོད་རྣམས་ལ་ཁྲུས་ཆོག་དང་མེ་ཐུན་བརྡེག་ ༣༤༠༽ པ་ལ་སོགས་བགེགས་སྦྱང་བའི་ཆོ་ག་རྣམས་བྱས་ནས་སྒྲུབ་ཁང་སྐོར་བ་བྱེད་ཅིང་ཡིག་བརྒྱ་བཟླས་བཞུག་ སློབ་དཔོན་དང་རྒྱལ་ཐེབས་སོགས་དཔོན་སློབ་རྣམས་ཀྱིས་སྒྲུབ་ཁང་ཐེམ་འོག་ཏུ་དམ་སྲི་མནོན་པའི་ལས་རིམ་དང༌། སྒོ་བྱང་ཕྱི་ནང་གདག་ཅིང་སོ་སོའི་སྐྱེད་ཆོག་དགུག་བསྟིམ་མཆོད་གཏོར་གཉེར་གཏད་རབ་གནས་དང་བཅས་གཞུང་བཞིན་བྱ། དེ་ནས་སློབ་མ་རྣམས་ནང་དུ་ཚོགས་ནས། དང་པོ་བྱུར་ཕུད་པ། བགེགས་སྦྱང་བ། ཌཀྐི་འགྱེད་པ། མཚམས་ཕུར་གདབ་པ་རྣམས་དང༌། ཕྱག་འཚལ། མཎྜལ། ཆོས་བཤད། སྐྱབས་འགྲོ་ སེམས་བསྐྱེད། སྡོམ་བཟུང༌། གནས་སུམ་བྱིན་རླབས། སོ་ཤིང་དོར་བ། ཁྱོར་ཆུ་སྦྱིན་པ། སྲུང་སྐུད་གདབ་པ། ཀུ་ཤ་སྦྱིན་པ་དང་བཅས་པའི་སློབ་མ་སྟ་གོན་གྱི་ཆོས་བཅོགས་སོ་རྣམས་དབང་གཞུངཆེན་མོ་སྨིན་བྱེད་སྙིང་གི་མེ་ལོང་ཉིད་ནས་ཇི་ལྟར་འབྱུང་བ་བཞིན་ཐར་ཆགས་སུ་བྱ་ཞིང༌། བླ་མས་ ༣༤༡༽ སྟ་གོན་གྱི་དགོས་ཆེད་དང་དག་སྦྱོར་བཤད་དེ་ཤིས་པ་བརྗོད་ནས་ཚོགས་པ་རྣམས་གྱིས་ལ། དཔོན་སློབ་རྣམས་ཀྱིས་དཀྱིལ་འཁོར་ས་ལ་བཏང་རག་གི་མཆོད་བསྟོད། ཚོགས་འཁོར། སྲུང་མ་སྤྱི་ལ་གཏོར་འབུལ་སོགས་བྱ་ཞིང༌། དཀྱིལ་འཁོར་གྱི་ཕྱོགས་བཞི་ནས་གསོལ་བ་གདབ་པ་དང༌། ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་ནམ་མཁར་བཏེགས་པའི་སྔགས་དང་ཕྱག་རྒྱ་ཏིང་འཛིན་བྱས་ནས། རྫས་དང་ཡོ་བྱད་ཐམས་ཅད་གཙང་སར་བསྡུའོ། དེ་ལྟར་འདི་ཡན་ཆད་རྣམས་སྔོན་འགྲོ་སྟ་གོན་གྱི་ལས་རིམ་མོ། དེ་ནས་ཞག་བཞི་པའི་ཚེ། དངོས་གཞིའི་ལས་རིམ་རྣམས་ཀྱི་དང་པོར། ཕྱི་ནང་གསང་གསུམ་གྱི་ཆག་ཆག་གདབ་ཅིང་ཐིག་ཆོག་རྒྱས་པར་བྱས་ཏེ་དཀྱིལ་འཁོར་གྱི་ཐིག་གཞུང་བཞིན་བཏབ། རྡུལ་ཚོན་བྱིན་གྱིས་བརླབ་ལ་དཀྱིལ་འཁོར་ནོར་འཁྲུལ་མེད་པར་བྲིས་ནས། ཡིག་བརྒྱས་ཁ་བསྐང་ཞིང་རོལ་མོས་ ༣༤༢༽བྱིན་ཕབས་པས་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་ནམ་མཁར་བཏེགས་པ་དེ་ཉིད་རྡུལ་ཚོན་དམ་ཚིག་གི་དཀྱིལ་འཁོར་ལ་བསྟིམ་མོ།

ཞག་ལྔ་པའི་ཚེ། ཐོག་མར་སློབ་དཔོན་གྱི་མདུན་དུ་གཞི་མ་གཙང་མ་རྒྱ་ཆེར་བཏིང་བའི་ཁར་མེ་ཏོག་བསིལ་མ་བཀྲམ་ནས། བུམ་པ་རྣམས་དང་སྐུ་རྟེན། དབང་རྫས། དཀྱིལ་འཁོར་རྒྱན་བཀོད་ཀྱི་རྫས་དང་ཡོ་བྱད་མཐའ་དག་གོ་རིམ་བཞིན་བཀྲམ་ལ། རོལ་དབྱངས་ཏིང་འཛིན་དང་བཅས་པས་སོ་སོའི་ཤར་གདབ་མངའ་གསོལ་གྱི་རིམ་པ་རྣམས་གྱེར་ཞིང༌། ལས་ཀྱི་རྒྱལ་ཐེབས་རྣམས་ཀྱིས་སོ་སོའི་གྱེར་ཚིག་དང་བསྟུན་པས་དབང་རྫས་སྒྲུབ་རྫས་སོགས་རིམ་པས་གདན་དྲངས་ནས་དཀྱིལ་འཁོར་དུ་དགོད་ཚུལ་ཕྱག་བཞེས་མཐོང་བརྒྱུད་ལྟར་བཞུགས་གསོལ་ཞིང་སྟེང་འོག་གི་རྒྱན་དགོས་ཀྱི་རིམ་པ་མཐའ་དག་ཀྱང་བརྗོད་ཚིག་དང་མཐུན་པར་དགྲམ། ཕྱི་རིམ་མཐའ་སྐོར་དུ་སྨན་ ༣༤༣༽ རཀྟ་གཏོར་མཆོད་སོགས་གྱ་ནོམ་པ་བཀོད་པའི་ཁྱད་པར་ཕུན་སུམ་ཚོགས་པས་རྩ་གསུམ་ལྷ་ཚོགས་ཆོས་སྲུང་ཐམས་ཅད་མཉེས་ཤིང་སྒྲུབ་པའི་མཆེད་ལྕམ་ཀུན་ཀྱང་སྣང་བ་འགྱུར་བའི་བཤམ་བཀོད་དང་ལྡན་པར་བྱའོ། ཞར་ལ་སློབ་དཔོན་མཆེད་ལྕམ་ཀརྨ་གིང་ཆེན་སོགས་ལ་མངའ་གསོལ་བའི་རྟགས་ཀྱི་ཕྱག་རྒྱའི་ཡོ་བྱད་རྒྱན་ཆས་ཐམས་ཅད་ཀྱང་མ་ཚང་བ་མེད་པར་དཀྱིལ་འཁོར་གྱི་ཉེ་ལོགས་གར་འོས་ཀྱི་སར་བཤམས་ལ་བཞག་གོ་ ༈ དེ་ནས་མཆོད་རྫས་བསང་སྦྱང་བྱས་ཏེ། སྐྱབས་སེམས་བདུན་དང་སྲུང་འཁོར་མཚམས་གཅོད་ཀྱི་བར་རྣམས་གྲུབ་པ་དང་སློབ་མ་རྣམས་ཚོགས་པ་ལ་བླ་མས་དལ་འབྱོར་རྙེད་དཀའ་མི་རྟག་ལས་འབྲས་འཁོར་བའི་སྡུག་བསྔལ་ཐར་པའི་ཕན་ཡོན་སོགས་དང༌། སྒོས་ལམ་ཁྱད་པར་དུ་ཟབ་ཆོས་དགོངས་པ་ ༣༤༤༽ འདུས་པའི་ཆེ་བ། བླ་མ་རྒྱུད་པའི་ལོ་རྒྱུས་རྣམས་དང༌། ཡང་དགོས་སུ་སྒྲུབ་པ་ཆེན་མོའི་འབབ་ཚུལ་ཕུན་སུམ་ཚོགས་པ་ལྔའི་མཚན་ཉིད་དགོས་ཆེད་སྐབས་དོན་གྱི་རྣམ་བཤད་རྒྱས་བསྡུས་སྐབས་དང་སྦྱར་ཏེ་བཤད་པའི་མཐར། རྒྱལ་ཐེབས་རྣམས་ཀྱིས་སྟོན་པ་རྡོ་རྗེའི་སློབ་དཔོན་ཉིད་ལ་མངའ་གསོལ་བའི་རྒྱན་ཆས་ཡོ་བྱད་རྣམས་འཐོགས་ནས། དེའི་གྱེར་ཚིག་མ་བུ་གོང་འོག་སྤྲད་ཚུལ་རྣམས་གཞུང་བཞིན་མ་ནོར་བར་བྱས་ལ་ཞི་ཆོས་ཁྲོ་ཆས་ཡོངས་རྫོགས་དང༌།

གཞན་ཡང་དབང་དགོས་ཟབ་བེར་པད་ཞྭ་སོགས་མཐའ་དག་དབུལ་ཞིང་མངའ་གསོལ་ཕྱག་མཆོད་བྱ། དེ་རྗེས་བཟུང་མ་བུ་འཛིན་གྱི་ལྷ་མི་དགུའམ། ལྔ་ལ་སོགས་པ་རྣམས་ལ་ཡུམ་ཀུན་ཏུ་བཟང་མོ་སོགས་སུ་མངའ་གསོལ་བའི་སྐབས་སུ། དཔོན་སློབ་རྣམས་ཀྱིས་རོལ་དབྱངས་གཡབ་རྫས་སྤོས་དུད་སོགས་ ༣༤༥༽ འཛོམ་ཞིང་ཏིང་འཛིན་རྣོན་པོ་དང་ལྡན་པས། མཁའ་འགྲོའི་ལས་བྱང་པདྨའི་ཟེ་འབྲུ་ནས་འབྱུང་བའི་ཡེ་ཤེས་མཁའ་འགྲོ་སྤྱན་འདྲེན་ནན་ཏན་ཏུ་བྱས་བསྟིམ་པས་དམ་ཚིག་གཙང་ཞིང་སློབ་དཔོན་ནུས་པ་དང་མཁའ་འགྲོའི་དམ་རྫས་ཚོགས་ན་མཁའ་འགྲོ་སྤྱན་འདྲེན་ནན་ཏན་ཏུ་བྱས་བསྟིམ་པས་དམ་ཚིག་གཙང་ཞིང་སློབ་དཔོན་ནུས་པ་དང་ལྡན་དང་མཁའ་འགྲོའི་དམ་རྫས་ཚོགས་ན་མཁའ་འགྲོ་དངོས་སུ་བབས་པས་ཉེས་སྨྲ་བ་ཡང་འབྱུང་ལ་འདི་སྐབས་སུ་གཏེར་སྟོན་ཉིད་ཀྱི་ཕྱག་བཞེས་མན་ངག་ཏུ། བུམ་འཛིན་གྱི་གཙོ་མོ་དང་ཕྱོགས་ཀྱི་བུམ་འཛིན་གང་ལ་བབས་འགྱུར་གྱི་མིང་བྱང་གི་རྟེན་འབྲེལ་བརྟག་པའི་ཐབས་ངེས་འགྲོངས་ཡོད་མོད་ཀྱང༌། དིང་སང་ནི་འདི་དག་གི་ལག་ལེན་གནད་ལ་མཁལ་ནུས་ཀྱི་དཔོན་སློབ་སྣོད་བཅུད་ནུས་པ་མཉམ་པ་མི་རྙེད་ཚེ། ཚུལ་དེ་དག་གི་བྱ་ལུགས་ཞིབ་ཏུ་བརྗོད་དགོས་པར་གྱུར་མོད། ༈ དེ་དག་གོང་གི་འཕྲོར། གཟུངས་མ་ལ་ཞིབ་ཆས་ཀྱིས་མངའ་ ༣༤༦༽ གསོལ་བ་དང༌། གིང་པ་རྣམས་ལ་གིང་ཆས། གཞན་ཡང་རྒྱལ་ཐེབས་རིགས་སོ་སོ་མཆེད་ལྕམ་དྲལ་ལས་བྱེད་འདུས་པ་ཀུན་ལ་མངའ་གསོལ་སྤྱི་དང་བྱེ་བྲག་ཐམས་ཅད་ཚུལ་བཞིན་བྱ་ཞིང༌། དམ་ཚིག་རྟགས་ཀྱི་རྒྱས་གདབ་པ་རྣམས་ཀྱང་འདིར་བྱའོ། ༈ དེ་ནས་དཔོན་སློབ་ཀུན་ཆས་ཀྱིས་བརྒྱན་ཏེ་གར་འཆམས་སྤོས་རོལ་མཆོད་རྫས་ཀྱི་ཡོ་བྱད་སྣ་ཚོགས་འཐོགས་པས། དཀྱིལ་འཁོར་བསྐོར་ཞིང་བརྡ་སྒོ་དབྱེ་བ། དུར་ཁྲོད་བརྒྱད་ཀྱི་བྱིན་འབེབས་སོགས་ཀྱང་སྐབས་དང་བསྟུན་པས་བྱའོ། སླར་ཡན་ལག་བདུན་དང་ཕྱི་ནང་གི་མཆོད་པ་བྱིན་རླབས་སོགས་ནས་བཟུང་སྟེ་བདག་མདུན་བུམ་གསུམ་བསྐང་གསོ་ཚོགས་ལས་སོགས་ཡོངས་སུ་རྫོགས་པར་གཞུང་བསྲང་ངོ༌།

༣༤༧༽ ཞག་དྲུག་པའི་ཚེ་ཐོག་མར་བདག་མདུན་བུམ་གསུམ་མཆོད་བསྟོད་རྣམས་འགྲུབ་པར་བྱས་ལ། སློབ་མའི་ཚོགས་ལ་དཀྱིལ་འཁོར་ཕྱི་འཇུག་ནང་འཇུག་གི་ལས་རིམ་རྣམས་དབང་གཞུང་ཆེན་མོ་ཉིད་ནས་འབྱུང་ལྟར་བྱ་བ་སྟེ། ཁྱད་པར་ཡེ་ཤེས་འབེབས་པའི་ཚེ་རྩ་བ་གསུམ་སོ་སོའི་སྤྱན་འདྲེན་ལས་བྱང་ཚངས་པའི་སྒྲ་དབྱངས་ལྟར་དང༌། དམིགས་རིམ་ནི་བྱིན་སློབས་གཏེར་མའི་བརྡའ་དབང་གི་གཞུང་དུ་ཡོད་པ་ལྟར་ནན་ཏན་དུ་འབོག་ཅིང་བསྟིམ། དམ་ཚིག་འབོག་པའི་དུས་སུ་ནི་དམ་ཚིག་རྣམ་བཤད་བརྒྱ་མདུད་ཀྱི་གཞུང་བཞིན་དང༌། དཀྱིལ་འཁོར་གྱི་བརྡ་དོན་དག་སྦྱོར་ནི་དཀྱིལ་འཁོར་རྣམ་དག་རབ་བསལ་དང༌། འཛབ་ལ་འཇུག་པའི་སྔོན་དུ་ལྷ་སྔགས་ཀྱི་དོན་རྣམས་འགྲེལ་བ་མེ་ཏོག་རྒྱན་ནས་གསུངས་པ་ལྟར་བཤད་པར་བྱ་སྟེ། འདི་དག་ཐམས་ཅད་སློབ་དཔོན་ཆེན་པོའི་གསུང་དྲི་མ་མེད་ ༣༤༨༽ པ་དགོངས་འདུས་ཉིད་ཀྱི་རང་གཞུང་ཡིན་པའི་ཕྱིར། ཆོས་གཞན་ལས་དྲངས་དགོས་མ་མཆིས་སོ། ༈ དེ་ནས་འཛབ་ལ་འཇུག་ཚེ། འཛབ་དགོངས་དམིགས་པའི་འཁྲུལ་འཁོར་གྱི་གཞུང་ལྟར་དམིགས་པ་འབོག་ཅིང་ཏིང་འཛིན་བསླབ་བྱ་ཞིབ་ཏུ་བྱས་ལ། འཛབ་བསྐུལ་རྣམས་རིམ་པ་ལྟར་དབྱངས་དང་བཅས་པས་བསྐུལ་ཞིང་འཛབ་་རྒྱུན་གྱི་དབུ་འཛུགས་པར་བྱ་བ་ལས། དེ་ཡང་སྒྲུབ་པའི་ཞག་གྲངས་ཇི་ཙམ་བྱ་བའི་དང་པོའི་ཞག་ལ་ནི་གཙོ་མཆོག་གི་འཛབ་ཁོ་ན་ལ་འབད་ཅིང་ཐུན་མཚམས་ལ་མཆོད་བསྟོད་དང་བཅས་པས་སྒྲུབས། ཞག་གཉིས་པ་ལ་འཁོར་གྱི་གཙོ་བརྒྱད་སོ་སོའི་འཛབ་བསྐུལ་དང་བཅས་ཏེ་སྒྲུབས། དེ་ལྷག་ཞག་གྲངས་ཇི་ཙམ་བྱས་པ་ཐམས་ཅད་སྤྱི་དྲིལ་ཆེན་མོ་འབའ་ཤིག་ལ་

འབུངས། ཕྱོགས་མཚམས་ཀྱི་གཟུངས་ཐག་ ༣༤༩༽ འཛིན་པའི་སློབ་དཔོན་བརྒྱད་པོས་ནི་གཙོ་བོ་བརྒྱད་པོའི་ཏིང་འཛིན་དང་བཅས་པ་བྱ། དེ་ལྟར་མི་ལག་གིས་མ་ལྕོགས་ན་གང་ཡོད་ཀྱི་སྤྱི་དྲིལ་ཆེན་མོ་ལ་འབད་ཅིང༌། སྟོན་པ་རྡོ་རྗེ་སློབ་དཔོན་ཉིད་ཀྱིས་ནི་དེ་བཞིན་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་གྱི་ཀློང་ནས་རྡོར་བཟླས་དང་སྤྱི་དྲིལ་སྤེལ་མར་བྱ། མཆེད་ལྕམ་ལ་ལས་ནི་དཀྱིལ་ཆོག་གི་གཞུང་བསྲང་ཞིང་ལྷ་ཚོགས་ལ་མཆོད་བསྟོད་དང་བཀའ་སྲུང་གཏེར་སྲུང་རྣམས་ལ་བསྐང་གསོ་ཚོགས་མཆོད་སོགས་རྒྱུན་མི་འཆད་དུ་འབད། རྡོ་རྗེ་རྒྱལ་ཐེབས་འཁོར་བཅས་ཀྱི་ཉིན་མཚན་གྱི་ཐུན་དྲུག་ཏུ་གིང་གཏོར་འབུལ་ཞིང་གིང་བསྐུལ་བྱས་ནས། གིང་རྣམས་ཀྱི་མྱུལ་ལེ་བསྐྲད་པའི་ལས་རིམ་རྣམས་ཀྱང་ཕྱག་ལེན་མཐོང་བརྒྱུད་ལྟར་དུས་ལས་མི་ཡོལ་བར་བྱ་བ་སོགས། མདོར་ན་སྒྲུབ་པ་དབུ་བཙུགས་ནས་ ༣༥༠༽ དབང་ཞལ་ཕྱེའི་བར་དུ་འཛབ་རྒྱུན་སྐད་ཅིག་ཀྱང་ལམ་ཆད་པ་དང༌། བསྐང་རྒྱུན་སོགས་ལེ་ལོའི་སེང་ཅུང་ཟད་ཀྱང་མ་ཤོར་བའི་བསྒྲུབ་པར་བྱའོ། གལ་ཏེ་བདུད་རྩི་སྨན་གྱི་སྒྲུབ་པ་དང་སྦྲེལ་ན། འདི་སྐབས་སུ་སྔོན་ལ་གོང་བཞིན་རྩ་བའི་སྒྲུབ་པ་ཞག་བདུན་ནམ་ལྔའམ་གསུམ་སོགས་གང་འོས་བྱས་ལ། དེ་ནས་ཁྲོལ་བུ་སྒྲུབ་པའི་བཅས་བཤམ་སྟེ་སྨན་སྒྲུབ་བདུད་རྩིའི་རྒྱ་མཚོའི་གཞུང་ཉིད་ནས་གསུངས་པའི་བསངས་སྦྱངས་བཞུ་བཏུལ་སོགས་ཀྱི་ལས་རིམ་ཚུལ་ལྡན་དུ་ཁྲོལ་སྒྲུབ་ཞག་གསུམ་སོགས་སྐབས་བསྟུན་དུ་བྱ་ཞིང༌། སླར་སྨན་རྣམས་བཏགས་བཙགས་སྦོལ་བའི་རིམ་པ་རྣམས་ཀྱང་ཕྱག་ལེན་མ་འཆུག་པ་གཞུང་དང་མཐུན་པར་བྱས་ལ་གཞལ་ཞིང་རྒྱས་གདབ་མངའ་གསོལ་གྱི་ལས་རིམ་རྒྱས་པ་དང་བཅས་པས་དཀྱིལ་འཁོར་དུ་བཞུགས་སུ་གསོལ་ནས་དུག་གསུམ་གྱི་ ༣༥༡༽ མེ་སྦར་ནས། སྟེང་ནས་སྐུ་གསུམ་གྱི་དར་ཁང་དབུགསལ་སོགས་བྱ་སྟེ། ལར་བདུད་རྩི་སྨན་གྱི་སྒྲུབ་པ་འདི་ལ་ནི་འོལ་ཚོད་རིག་དཔྱོད་ཀྱིས་་འགྲུབ་པའི་སྤྲུལ་མ་ཡིན་ཚེ།

བླ་མ་བརྒྱུད་པའི་ཞལ་གདམས་ཕྱག་བཞེས་མངོན་སུམ་དུ་གཏན་ལ་ཕེབས་པ་དང༌། ཉམས་མྱོང་གི་དྲོད་ཐཽབ་པ་གལ་ཆེ་བ་ཡིན་པ་ཤེས་པར་བྱ་ཞིང༌། ཉིན་མཚན་དུས་དྲུག་ཏུ་རྟ་བྲོ་བགེགས་བསྐྲད་པ་དང༌། སྨན་གྱི་ལྷ་མོ་སྤྱན་འདྲེན་བྱིན་འབེབས་ཀྱི་སྐབས་ཐམས་ཅད་དུ་རྡོ་རྗེ་རྒྱལ་པོ་རྒྱལ་ཐེབས་བུམ་འཛིན་གྱི་ལྷ་མོར་བཅས་པ་རྣམས་ཀྱིས་སྤོས་དང་རོལ་མོ་གླིང་བུ་གཡབ་དར་ལྔ་ལྡན་སོགས་འཛོམས་པས་གདུང་བའི་དབྱངས་ཀྱིས་བླ་མ་རིག་འཛིན་དྲང་སྲོང་ཡི་དམ་ལྷ་ཚོགས་སྨན་གྱི་ལྷ་མོ་འབྱུང་བའི་ལྷ་དང་སྲུང་མ་སོགས་སྤྱན་དྲངས་ཤིང་བསྟིམ་པའི་རྫས་སྔགས་ཏིང་འཛིན་ནན་ཏན་དུ་འབད་དེ་ ༣༥༢༽དཔོན་སློབ་ཀུན་དམ་ཚིག་གི་སེལ་མ་ཞུགས་ཅིང་ལེ་ལོ་བཏང་སྙོམ་ཚོད་ཡོད་གལ་མ་མཆིས་སུ་མི་བྱ་བར། མུ་གོས་གདུང་ཤུགས་བརྩོན་གྲུས་དྲག་པོས་སྨན་གྱི་འཛབ་དབྱངས་གསུང་མི་འཆད་དུ་སྒྲོག་ཅིང་རྩེ་གཅིག་ཏུ་སྒྲུབས། མདོར་ན་སྒྲུབ་པ་འདི་ལ་ནི་ཇི་སྲིད་རྟགས་མ་བྱུང་གི་བར་དུ་སྒྲུབ་པའི་འཕྲོ་བཤེལ་ཐབས་མེད་ཚེ། ཐོག་མ་ནས་དཔོན་སློབ་ཐམས་ཅད་དེ་ལྟ་བུའི་གོ་ཆ་དང་འདུན་པ་གཅིག་ཏུ་མཐུན་པ་གལ་ཆེའོ། དེ་ནས། འོད་དང་དུ་བ་སྒྲ་དང་དྲི། ལྷ་མོའི་གཟུགས་ཀྱང་སྟོན་པར་བྱེད། ཅེས་པ་ལྟ་བུའི་རྟགས་ཀྱི་རིམ་པ་སོ་སོའི་སྐལ་བ་དང་མཐུན་པ་ནཾ་བྱུང་བའི་ཚེ། ནང་པར་དངོས་གྲུབ་ལེན་རྒྱུའི་ནུབ་མོ་ནས་མཎྜལ་ཞལ་བསྒྱུར་གྱི་དཀྱིལ་འཁོར་དང༌། ཚེ་སྒྲུབ་བདུད་རྩི་སྙིང་པོའི་དབང་རྫས། དངོས་གྲུབ་ཀྱི་རྫས་དང་ཡོ་བྱད་ཐམས་ཅད་རྒྱས་པར་ནས་དེ་དག་གི་ཆོ་ག་དང་དངོས་གྲུབ་བསྐུལ་བའི་འཛབ་རྣམས་ལ་འབུངས་ ༣༥༣༽ཤིང་ལས་དབྱང་རྒྱས་པ་དང་བསྟོད་འབུམ་བསྐང་བ་ཚེ་འགུགས་ཚོགས་བརྒྱ་སོགས་དང༌། ཁྱད་པར་ལྷ་གནད་ལ་དབབ་སྟེ་དངོས་གྲུབ་འབེབས་པའི་ལས་རིམ་རྣམས་ཀྱང་སྨན་སྒྲུབ་ཀྱི་གཞུང་བཞིན་བྱ། ནང་པར་ཐོ་རངས་ནས་སྒྲུབ་པའི་སྤྲ་ཁྲིད་ལས་གསུངས་པའི་དངོས་གྲུབ་བླང་ཆོག་དང་བདུད་རྩི་སྨན་གྱི་དངོས་གྲུབ་བླང་ཆོག་སོགས་རྒྱས་པར་བྱས་ནས་དཔོན་སློབ་ཀུན་སྨན་བུམ་གྱི་དབང་བླང༌།

དེ་ནས་ནམ་ལངས་པ་དང་སྨན་གྱི་རྒྱ་དཀྲོལ་ཏེ། མཚན་ལྡན་གྱི་ཐོད་པར་སྨན་ཕུད་ཛཱ་གད་དང་སྦྱར་ལ་སྨན་མཆོད་རྒྱས་པར་བྱས་དངོས་གྲུབ་བླང་ཞིང་སྨན་བོག་བ་དག་ཚོགས་ཀྱི་དགའ་སྟོན་བྱའོ། ཉི་མ་ཤར་བ་དང་དུས་སྤྲད་དེ་བདུད་རྩི་སྙིང་པོའི་ཞཝ་དབང་བསྐུར་བ་དང༌། དེའི་ཉིན་སྒྲུབ་ས་ལུང་དབང་རྒྱས་པར་མཆད་པ་ཡང་གཏེར་སྟོན་གྱི་ཕྱག་བཞེས་སུ་གདའ། དེའི་ ༣༥༣༽ནང་སར་དངོས་གྲུབ་འཕེལ་བྱེད་རྒྱས་པའི་སྦྱིན་སྲེག་བྱ་ཞིང༌། ཉམས་ལེན་གྱི་སྣོད་ལྡན་རྣམས་ལ་ནི་གཏུམ་མོའི་མེ་དབང་དགའ་བ་འཁྱིས་པ་བདེ་བའི་གསང་མཛོད་ཀྱི་དབང་ཡང་སྐབས་འདིར་བསྐུལ་བར་མཛད་དོ།། དེ་ནས་དབང་དངོས་གཞི་སྤྲོས་བཅས་བུམ་དབང་ཆེན་མོ་ཉིད་སྐབས་ཐབ་དང་བསྟུན་དབུ་གཙུག་ལ། དེ་ཡང་རིག་པའི་དབང་ལྔ་མཐའ་རྟེན་དང་བཅས་པ་བསྐུར་བའི་དུས་སུ། སྟོན་པའི་རྡོ་རྗེ་རྒྱལ་པོ་རྒྱན་ཐམས་ཅད་དང་ལྡན་པ་བུམ་འཛིན་གྱི་ལྷ་མོར་བཅས་བས་དབང་རྣམས་བསྐུར་ཚེ་བྱང་སེམས་རྣམས་ཀྱིས་མེ་ཏོག་འཐོར་ཞིང་ཤིས་པ་བརྗོད་པ་དང༌། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་བླང༌། ལྷ་མོ་རྣམས་ཀྱིས་མཆོད་པ་འབུལ། ཏ་ཀྲྀ་ཏ་ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་སྐྲོད་པ་སོགས་ཀྱང་བརྗོད་བྱ་དང་མཚུངས་པའི་ཆས་ཞུགས་སོགས་ཕྱག་ལེན་མཐོང་བརྒྱུད་ལྟར་བྱ་ཞིང༌། དབང་གི་རིམ་པ་རྣམས་ཡུལ་ ༣༥༥༽དུས་གང་ཟག་གི་སྣོད་དང་ཇི་ལྟར་འཚམས་པར་སློབ་དཔོན་ཉིད་ཀྱིས་དགོགས་པས་དཔྱད་དེ་ཞག་གྲངས་དང་རྒྱས་བསྡུས་སྐབས་བསྟུན་དུ་བྱ་ལགས། དབང་གོང་མ་རྣམས་ཀྱིས་ཚེ་སློབ་མ་ཆོས་ཀྱི་རྒྱུན་འཛིན་གཞན་དོན་ནུས་པའི་རིགས་ལ་ནི། སྤྲོས་མེད་དོན་དབང་སྲེག་རྩ་དམར་འབྱིན་དབང་བཞིའི་ལག་ལེན་བདེ་སྟོང་རམ་འབྱམས། ཆགས་ཆེན་ལམ་སློངས་ཌཱ་ཀིའི་སྲོག་དབང༌། ཐོད་རྒལ་རྩལ་དབང་བརྡ་དོན་ཟུང་འབྲེལ། འཁོར་འདས་གཞི་རྫོགས་རང་བྱུང་དོན་དབང་ལ་སོགས་པའི་དབང་ཆེན་རྣམས་ཀྱང་ལྷན་ཐབས་སུ་མ་བུ་སྤྲད་ནས་བསྐུར་ཚུལ་དང༌།

དེ་བཞིན་དུ་གཅོད་དབང་སྒོ་འབྱེད། པུ་སྟིའི་ལུང་དབང་ལ་སོགས་དབང་ཕྲན་རྣམས་དང༌། གཞན་ཡང༌། བཏགས་གྲོལ་རྒྱུད་དབང་དང༌། གཅོད་དབང་བཀའ་རྒྱ་མ་དང༌། ལམ་འབྲས་ཀྱི་ཀྱིས་དབང་གུད་དུ་སྦས་པ་སོགས། ༣༥༦༽ ༣༣༧༽ཉིད་ཀྱི་རྒྱལ་ཚབས་སྟོན་པའི་ཞལ་སྐྱིན་སྟོན་པའི་སྒྲོན་མེར་གྱུར་ངེས་ལ་མ་གཏོགས༴ ཚོགས་ས་སྤྱི་དང་གང་ཟག་ཕལ་པ་དག་ལ་བསྐུར་དུ་རུང་བ་ནི་མ་ཡིན་ཕྱིར༴ གཙོ་བོར་སྣོ་དང་སྦྱོར་ཚུལ་ཤེས་དགོས་ལ༴ འདི་དག་བསྐུར་བའི་སྐབས་སུ་ནི། བཀའ་སྲུང་རྟ་ནག་སྡེ་བརྒྱད༴ གཏེར་སྲུང་འབར་བ་བླ་བདུན། ཆོས་ཀྱི་རྗེས་འབྲེང་ཤ་ཟ་ཁ་མོ་ཚེ་དང་བཅས་པ་རྣམས་ཀྱི་ལས་གཏོར་རྒྱན་དལན་གཟི་བརྗིད་འབར་བར་བཤམས་ནས༴ སོ་སོའི་གསོལ་མཆོད་རྒྱས་པ་ནན་ཏན་དུ་བྱེད་པ་ཡང་ཆོས་འཁྲེན་དང་རུ་དོག་མི་འབྱུང་བའི་ཐབས་གནད་གལ་པོ་ཆེ་ཡིན་པ་དང༌། སྲོག་དབང་ཆེན་མོའི་ཚེ་དྲེག་པ་སྡེ་བརྒྱད་སོ་སོའི་ཆོས་ཞུགས་རྔམ་པའི་འཁོར་ཚོགས་དང་བཅས་པ་སྟ་གོན་བྱས་ལ་དཔོན་སློབ་རྣམས་ཀྱི་སྤྱན་དྲེན་དྲག་ཏུ་བྱས་པའི་མཐར་རོལ་མོ་ ༣༥༨༽སྤྱུག་གླུ་རྐང་གླིང་བསུར་དང་བཅས་པས་སྤྱན་དྲངས་ཏེ་འགྱིང་འཚམས་རྔམས་བསྟབས་དང་ལྡན་པས༴ དབུ་མར་རྣམས་མ་རྫོགས་ཀྱི་རིང་ལ་དཔང་པོའམ་དམིགས་རྟེན་གྱི་ཚུལ་དུ་བཞུགས་སུ་གསོལ་ཞིང་ཤ་ཁྲག་ལ་གཞི་བླང་པའི་དགྲེགས་ཀྱི་ལིང་ཆེན་བཟོས་པ་ལ་དགུག་གཞུག་བསྐྲལ་བསྟབས་བྱ་ཞིང༴ གཏོཏ་མ་གསེར་སྐྱེམས་སོགས་རྒྱ་ཆེར་བུལ་ལ་ཕྲིན་ལས་གཞོལ་ནས་གཤེགས་གསོལ་བྱ་སྟེ༴ དེ་ལྟར་འདི་ཆད་ཐམས་ཅད་བསྒྲུབ་དབང་དངོས་གཞིའི་རིམ་པའོ། ༈ དེ་ནས་རྗེས་ཀྱི་ལས་རྣམས་ཀྱི་དང་པོར༴ དྲོག་མངོན་སྤྱོད་ལས་ལ་མཁོ་བའི་རྫས་དང་ཡོ་བྱད་རྣམས་གསར་ཞིང༴ ཡི་དམ་ཆོས་སྐྱོང་གི་གཏོར་མ༴ དྲག་པོའི་ཟོར་རྫས་སོགས་མཐའ་དག་ཚགས་སུ་ཆུད་ཅིང༌། ཕུང་གསུམ་བདུད་ཀྱི་ཉི་མ་ལ་བབས་པས་དུས་སུ༴ ཐོག་མར་བླ་མ་ཡི་དམ་མཁའ་འགྲོའི་ལས་བྱང་ཆོས་སྐྱོང་རྣམས་ཀྱི་བསྐང་གསོ་གཞུང་སྲང་

༣༥༩༽ཚོགས་ལས་བསྐང་བ་ལ་སོགས་རྒྱས་པ་སྔོན་དུ་བཏང་ལ༴ དྲག་སྔགས་ཕྲིན་ལས་ཀྱི་རྒྱུད་སྤུ་གྲི་རེག་ཆོད་དང་ཞི་ཁྲོ་བཀའ་བརྒྱད་སོགས་ལས་འབྱུང་བ་ལྟར་གྱི་མཁྱེན་གོལ་གཟུགས་དཔགས་ཐུགས་དམ་བསྐུལ་བ་རྣམས་དང་བསྒྲལ་སོགས་ལས་བྱང་ཚངས་སྒྲ་ལྟར་རྒྱབ་པར་བྱ་ཞིང༌། བསྐུལ་འབུམ་མྱུར་འགྱེགས་གླེག་འཕྲེང་དང༌། བཀའ་སྲུང་སྤྱི་དང་བྱེ་བྲག་པའི་བསྐུལ་བྱང་གི་ལས་རིམ་ཐམས་ཅད་གྲུབ་དང༌། ཕྱི་མཚམས་བསྲུང་བའི་རྒྱལ་ཆེན་རྣམས་ལ་གསེར་སྐྱེམས་དང་དཀར་གཏོར༴ བར་མཚམས་སྒོ་བྱང་གི་ཁྲོ་བོ་གཉིས་དང༌། ནང་མཚམས་བསྲུང་བའི་ཁྲོ་བཅུ་རྣམས་ལ་སྨན་རཱཀདང་བཅས་པས་བཏང་རག་གི་མཆོད་བསྟོད་ཕུལ་ཞིང་རང་གནས་སུ་གཤེགས་སུ་གསོལ་བ་སྒྲུབ་པ་སྤྲ་ཁྲིད་ལས་བྱུང་བ་ལྟར་བྱས་ནས་སྐུ་རྟེན་རྣམས་དཀྲོལ་ལ་ཐེ་སེར་རྗེས་མི་མངོན་པར་ལ་ཤིག ༣༦༠༽༈ དེ་ནས་དཔོན་སློབ་མཆེད་ལྕམ་ཐམས་ཅད་དྲག་པོའི་ཆས་སུ་ཞུགས་ཏེ། ཐོག་མར་ལས་བྱཧང་མེ་ཏོག་འཕྲེང་མཛེབ་ཀྱི་བསྐུལ་ཐུང་གི་བཞེས་སུ་གསོལ་ནས། ཆོས་སྐྱོང་གི་རུ་དར་དང་མེ་ཡིས་སྣ་དྲངས་ལ། ལས་གཏོར་རྣམས་ཕྱིར་དཔག་པ་དང་ལྷན་ཅིག་པར་དཔོན་སློབ་རྣམས་ཀྱིས། བཀའ་བརྒྱད་ནས་འབྱུང་བའི༴ ཤར་ཕྱོགས་དུང་སྒོ་དཀར་པོ་ཕྱེས། ཞེས་སོགས་ཀྱིས་ཟེར་ལམ་དབྱེ་ཞིང༌། ཕྱི་རོལ་དུ་སླེབ་པ་དང་གཞི་བདག་ཕྱོགས་སྐྱོང་རྣམས་ལ་གསེར་སྐྱེམ་ཕུལ་ནས་གདོང་གྲོགས་བཅོལ་ཞིང་འགྱུ་ཕྱོགས་སོགས་དང་མཐུན་པའི་ཕྱོགས་སུ་ཁ་བལྟས་ལ། འཆད་མདོའི་འབུལ་གཏོར་ཟུར་དུ་བྱས་ལ་ཕུལ་ཞིང་སྔོན་གྱི་མཐའ་ཚིག་གིས་འཆད་མེད་གྱེར༴ སླར་ཡང་རྡོ་རྗེ་རྒྱལ་པོས་སྣང་སྲིད་ཀྱི་ལྷ་མ་སྲིན་རྣམས་ལ་སྲིད་པ་ཧུའི་ཕྱག་རྒྱ་བསྟན་ཞིང༌། འདི་ནི་སངས་རྒྱས་གླིང་པའི་གཏེར་ཁ་ལས་བྱུང་བ་དྲ་ལས་མ་ཆེད་དུ་ ༣༦༠༽ བསྔགས་སོ། བཞེས་འབོད་དང་དམ་བདར་བྱས་ནས། ལས་བྱང་ཆེན་མོ་དང་རྟ་མགྲིན་ནག་པོའི་ལས་གཞུང་སོགས་ནས་བྱུང་བ་ལྟར་གྱི་གཏེར་གཟེར་གྱེར་ལ་གཏོར་མ་རྣམས་འཕང༌།

བསྟན་སྐྱོང་དང་གཏོར་བྲོ་སོགས་སྤྱི་ལྟར་བྱས་ཏེ་ནང་དུ་ལོག་ཚེ་དཀྱིལ་འཁོར་ལྷ་ཚོགས་ལ་བཏང་རིག་མཆོད་བསྟོད་རྒྱ་ཆེར་བྱས་པའི་མཐར༴ དངོས་གྲུབ་བླང་བའི་སྐབས་འདིར་བྱིན་རླབས་གཏོར་མ་བརྡ་དབང་ཉིད་མཐའ་རྟེན་གྱི་ཚུལ་དུ་བསྐུར་བ་ཡང་རིག་འཛིན་བརྒྱུད་པའི་ཕྱག་བཞེས་སོ། ༈ དེ་ནས་དཀྱིལ་འཁོར་གཤེགས་སུ་གསོལ་བསྡུ་ཆོག༴ སྔོ་སྨོན་ཤིས་བརྗོད་སོགས་རྗེས་ཆོག་མཐའ་དག་སྤྱི་དང་མཐུན་པར་གྲུབ་ནས༴ ཡོ་བྱད་ཐམས་ཅད་བསྡུས་པའི་ཤུལ་གྱི་སྟེགས་བུའི་ཁར་རྡོ་རྗེ་སློབ་དཔོན་བཞུགས་པའི་ཁྲིའུ་བཤམས་པ་ལ་བཞུགས་སུ་གསོལ། སློབ་མ་ཀུན་གྱིས་སྐུ་དྲིན་བཏང་རག་གི་མཆོད་གསོལ་ ༣༦༢༽འབུལ་ནོད་དང་བཅས་ཏེ། སྒྲུབ་ཁོག་གསུངས་པའི་སྐུ་དྲིན་གསོ་ཞིང་སྨོན་ལམ་གདབ་པའི་ཚིག་རྣམས་དང་ཉེས་བྱས་བཤགས་པ་ལ་སོགས་ནན་ཏན་བྱ༴ མཆེད་ལྕམ་འདུ་ཚོགས་གང་ཡོད་ཀུན་གྱི་ལག་ཏུ་མར་མེ་བཟུང་ནས་དཔོན་སློབ་ཀུན་གྱི་སྨོན་ལམ་དང་གཞན་ཡང་སྨོན་ལམ་ཤིས་བརྗོད་ཀྱི་རིགས་རྒྱ་ཚེན་པོ་བྱའོ། ༈ འདི་སྐབས་སུ་སྒྲུབ་རྫས་བྱིན་རླབས་དངོས་གྲུབ་ཀྱི་རྫས་སྣ་ཚང་བ་དང་དར་སྨོན་རིན་ཆེན་འབྲུ་སྣ་སོགས་བཀྲ་ཤིས་པའི་ནུམ་གཏེར་གྱི་ཡོ་བྱད་ཚུལ་ལྡན་སྤྱི་དང་མཐུན་པར་བྱས་པའི་བུམ་པ་ལྔའམ་གཅིག་གིས་རུང་བ་ལ་རྒྱས་བཏབ་སྟེ་རབ་ཏུ་གནས་ཤིང་སེམས་སྐྱེད་སྨོན་ལམ་གྱིས་མཚམས་སྦྱར་བར་དཀྱིལ་འཁོར་གྱི་ཤུལ་དུའམ་གར་འོས་ཀྱི་སར་སྦ་ཞིང་ཚགས་བལྡམ། དེ་ནས་ནང་དཀྱིལ་འཁོར་གྱི་རྡུལ་ཚོན་རྣམས་རིན་ཆེན་གྱི་བུམ་པ་རྒྱལ་ལྡན་དུ་བརྩལ་བ༴ གཞན་ཡང་བུམ་ ༣༦༣༽ཆུ། བདུད་རྩི་ཆོས་སྨན་གྱི་ཆ་ཤས་ཅུང་ཟད་དང༌། དངོས་གྲུབ་ཀྱི་གཏོར་མའི་དུམ་བུ་སྣ་ཚང་ཐོད་པ་མཚན་ལྡན་ནམ་རིན་པོ་ཆེའི་སྣོད་དུ་བླུགས་གདན་དྲངས་ཤིང༌། ཚོགས་ཀྱི་ཡོ་བྱད། སྤོས་དང་རོལ་མོའི་བྱེ་བྲག དར་འཕན་མཆོད་རྫས་སེར་སྦྲེང་དང་བཅས་ན་ཆུ་ཀླུང་ཆེན་པོ་རྒྱུན་འབབས་ཀྱི་འགྲམ་ངོགས་སུ་འཁྱེར་ལ༴

དེར་དཔོན་སློབ་རྣམས་ཀྱིས་བདག་སྐྱེད་ལས་གཞུང་མདོར་བསྡུས་ཚོགས་མཆོད་དང་བཅས་པ་བྱས་པའི་མཐར། རྫས་དེ་དག་ཆུ་བོར་གཤེགས་པ་དང་ལྷན་ཅིག་ཏུ་ཡིག་བརྒྱ་སོགས་ཀྱི་གཟུངས་སྔགས་བྱིན་རླབས་ཅན་འུར་དིར་དུ་བསྒྲག་ཅིག༴ བསྟན་འགྲོའི་དོན་ཆེན་དང་ཆོས་རྒྱུད་མི་ནུས་རྒྱ་ཆེར་འབྱུང་བའི་སྨོན་ལམ་གདབ༴ བུམ་པ་སོགས་ཀྱི་སྣོད་རྣམས་སྟོང་པར་མི་ལོག་པར་ཆུ་གཙང་གི་བཀང་ནས་འཁྱེར་ཏེ་སྒྲུབ་ཁང་སོགས་ ༣༦༤༽ལ་འཐོར། དེ་རྗེས་ཞག་གསུམ་སོགས་ཀྱི་བར་གནས་བསྲུང་བ་དང༌། དགའ་སྟོན་གྱི་རིགས་ཀྱང་ཅི་འབྱོར་བྱ་བ་ནི་རྗེས་འཇུག་བཀྲ་ཤིས་པའི་རྟེན་འབྲེལ་གྱི་གནད་དུ་ཆེ་བར་གསུངས་སོ། དེ་ལྟར་ཟབ་ཆོད་དགོངས་པ་འདུས་པའི་སྨིན་བྱེད་སྒྲུབ་དབང་ཆེན་མོ་དངོས་གྲུབ་སྨན་དང་བྲལ་བའི་སྤྲོས་བཅས་ཚོམ་བུ་ཚོགས་སྒྲུབ་ཀྱི་སྒོ་ནས། རང་གཞན་སྨིན་གྲོལ་དུ་བྱ་བའི་ལས་རིམ་སྔོན་འགྲོ་དངོས་གཞི་རྗེས་དང་བཅས་པའི་ཕྱག་བཞེས་ཀྱི་ཁྲིག་རིམ་མཐར་ཆགས་ལྟས་མཆོག་ཏུ་ཟིན་ཎིས་སུ་བཏབས་པ་འདི་ནི། གཙོ་བོར་དགོངས་འདུས་ཉིད་ཀྱི་སྒྲུབ་ཁོག་ཆེན་མོ་སྤྲ་ཁྲིད་གསལ་བའི་མེ་ལོང༴ སྤྲོས་བཅས་བུམ་དབང་ཆེན་མོ། ལག་ཁྲིགས་གབ་པ་སྔོན་མངོན་ཕྱུངས༴ སྨན་སྒྲུབ་བདུད་རྩིའི་རྒྱ་མཚོ་སོགས་ལས་ཇི་ལྟར་གསུངས་པ་བཞིན་དང༌། ཕྱག་ལེན་གྱི་རིགས་ཕྲན༴ ༣༦༥༽བུ་ཁ་སྐོངས་དཀྲུག་སྡེབས་ཀྱི་རྣམ་པ་དག་ཀྱང་རིག་འཛིན་ཆེན་པོ་སངས་རྒྱས་གླིང་པ་ཉིད་ནས་ཞལ་གྱི་བདུད་རྩི་ཟམ་མ་ཆད་པ་ལས་ལེགས་པར་འོངས་པ་ཁོ་ན་མ་གཏོགས་གཟུ་ལུམ་རས་ཆོད་ཀྱི་དྲི་མའི་བསླད་པ་ཅུང་ཟད་ཙམ་ཡང་མ་མཆིས་ལགས་པས། ཡིད་བརྟན་གྱི་ཕྱི་མོར་རུང་བ་ནོར་བུའི་ཕྲེང་བ་ལྟ་བུ་འདི་ནི༴ རིག་འཛིན་ཆེན་པོ་སངས་རྒྱས་གླིང་པའི་རིགས་དང་ཆོས་ཀྱི་རིང་ལུགས་དྲི་མ་མེད་པ་འཛིན་པར་རྡོ་རྗེའི་སློབ་དཔོན་རིག་འཛིན་སྙིང་པོའི་སྤྱན་སྔར། བདག་ཉིད་དུས་ཀྱི་བླ་མའི་ཟོལ་གྱིས་འཚོ་བ་ས་མཁས་མཆོག་པདྨ་ལེགས་གྲུབ་རྩལ་གྱིས་ཡེ་གེའི་བྱེད་པོ་དང་བསྐུལ་བ་པོར་བགྱིས་ཤིང༌།

རྩེ་ལེ་རིག་འཛིན་ཡབ་སྲས་ནས་བརྒྱུད་པ་སོགས་རང་ལུགས་ཀྱི་བླ་མ་བརྒྱུད་པ་རྣམས་ཀྱི་ཕྱག་བཞེས་ཞལ་རྒྱུན་དག་དང་ཡང་དགོངས་པ་བསྟུན་ཏེ་ ༣༦༦༽བདག་ཅག་དཔོན་སློབ་ལྷན་གཅིག་ཏུ་གོ་བདུར་ནས་ཕྱི་རབས་ཀྱི་གང་ཟག་གསར་བུ་དག་ལ་ཕན་པར་བསམས་ཏེ་ཡི་གེར་བཀོད་པ་འདིས་མཚོན་ཆེན་རབ་ཀུན་གྱི་དགེ་ཚོགས་དུས་གསུམ་བསགས་པའི་རྣམ་དཀར་མཐའ་དག་གཅིག་ཏུ་བསྡོམ་པའི་མཐུ་འབྲས་ཀྱིས། མ་རྒན་མཁའ་མཉམ་འགྲོ་རྣམ་མཁྱེན་སངས་རྒྱས་ཀྱི་གོ་འཕངས་ལ་མ་འཁོད་དེ་སྲིད་དུ་ཐེག་ཆེན་བླ་མེད་རྡོ་རྗེ་ཐེག་པའི་བསྟན་པ་རིན་པོ་ཆེ་འགྲིབ་མེད་གོང་ནས་གོང་དུ་འཕེལ་ཞིང་རྒྱས་པའི་ཆ་རྐྱེན་དུ་གྱུར་ཅིག མངྒ་ལམ་བྷ་ཝནྟུ།།
