༤༧༥༽༄༅། རྟའི་རྒྱལ་པོ་བ་ལ་ཧའི་རྟོགས་པ་བརྗོད་པ་ལེགས་བཤད་ཆོས་ཀྱི་རྔ་སྒྲ་ཞེས་བྱ་བ་བཞུགས་སོ། ༤༧༦༽རྟའི་རྒྱལ་པོ་བ་ལ་ཧའི་རྟོག་པ་བརྗོད་པ་ལེགས་བཤད་ཆོས་ཀྱི་རྔ་སྒྲ་ཞེས་བྱ་བ་རྒྱལ་བ་སྲས་དང་བཅས་པ་ལ་གུས་པས་ཕྱག་འཚལ་ལོ། སྲིད་པའི་རྒྱ་མཚོ་མཐའ་མེད་བརྒལ་དཀའ་བ། །སྡུག་བསྔལ་བཟོད་དཀའ་ལས་ཉོན་རྦ་རླབས་ཀྱིས། །དལ་འབྱོར་གྲུ་བོ་དབང་མེད་གཡེང་རྣམས་ལ། །དེ་ལས་བཟློག་ཐབས་ཚིག་འགའ་བསྙད་པར་བྱ། །གང་ཞིག་འཁོར་བའི་རྒྱ་མཚོ་འཇིགས་རུང་བར། །ཡང་ཡང་འཁོར་ཏེ་ཟོ་ཆུའི་ཁྱུད་མོ་བཞིན། །གཏན་དུ་འཁྱམ་འདི་ད་ནི་བརྩོན་པའི་མཐུས། །འདི་ལས་མྱུར་དུ་ཐར་ལ་འབད་འཚལ་ལོ། ༤༧༧༽།དཔེར་ན་རྒྱ་མཚོ་ཆེན་པོར་རུས་སྦལ་ཞིག་གནས་ཤིང༌། །གཉའ་ཤིང་བུ་ག་གཅིག་པ་རླབས་ཀྱི་བསྐྱོད་བྱེད་པར། །རུས་སྦལ་མགྲིན་ཆུད་སྲིད་ཀྱི་དེ་བས་དལ་འབྱོར་ནི། །རབ་ཏུ་བརྙེད་པར་དཀའ་བ་མཁྱེན་མཛོད་གྲོགས་པོ་དག །གང་ཞིག་མེ་ཏོག་ཨུ་དུམ་ཝ་ར་དེ་ཉིད་ནི། །བརྒྱ་ལམ་ན་ཡང་རེས་འགའ་འབྱུང་བར་སྲིད་མོད་ཀྱི། །དལ་འབྱོར་མི་ལུས་རིན་ཆེན་དེ་བས་བརྙེད་པ་དཀའ། །གང་དེ་ཤེས་ལྡན་རྣམས་ཀྱི་ཐུགས་ལ་མཁྱེན་པར་མཛོད། །གང་ཞིག་དམྱལ་བ་ཡི་དྭགས་དུད་འགྲོ་ཚེ་རིང་ལྷ། མཐའ་ཁོབ་ ༤༧༨༽ཀླ་ཀློ་ལོག་ལྟ་སངས་རྒྱས་ཀྱིས་སྟོངས་དང༌། །ལྐུག་པ་ལ་སོགས་མི་ཁོམ་གནས་བརྒྱད་རབ་སྤངས་ཏེ། །དལ་འབྱོར་ཕུན་ཚོགས་ཤིན་ཏུ་ཐོབ་དཀའ་མཁྱེན་པར་མཛོད། །དེ་ལྟར་གལ་ཏེ་མི་ལུས་ཐོབ་གྱུར་ཀྱང༌། །ཚེ་འདི་མི་རྟག་ཆུ་ཡི་སྦུ་བ་འདྲ། །མྱུར་དུ་འཇིགས་ཤིང་འགྱུར་བའི་ཆོས་ཅན་དེ། །འདུས་བྱས་རྟག་པ་མ་ཡིན་མཁྱེན་པར་མཛོད། །གང་ཞིག་དབྱར་དགུན་སྟོན་དཔྱིད་དུས་བཞིའི་འགྱུར་ལྡོག་དང༌། །ཡུལ་དང་གྲོང་ཁྱེར་དགོན་གནས་ལ་སོགས་འཇིག་གྱུར་ཅིང༌། །དེ་ན་གནས་པའི་འགྲོ་བ་ཀུན་ཀྱང་འཆི་འགྱུར་བའི། །དོན་དེ་སོམ་ལ་ད་ལྟ་ཉིད་དུ་ཆོས་མཛོད་ཅིག །གང་ཞིག་བདག་གི་ཕ་མ་སྤུན་ཟླ་གྲོགས་དག་དང༌། །འཛའ་བཤེས་གཉེན་དང་ལོ་མཉམ་འཆི་བའི་ཚུལ་སོམས་ལ། །ཞག་རེ་སོང་ཞིང་འཆི་ལ་ཕར་ཕར་ཉེ་བའི་ཕྱིར། །མྱུར་བ་ཉིད་དུ་དམ་པའི་ཆོས་ལ་འབུངས་རེ་རན།

༤༧༩༽།གཞན་ཡང་རང་ཅག་མི་ལུས་མཆོག་ཐོབ་ནས། །དགེ་ལ་མི་བརྩོན་སྡིག་ལ་འཇུག་པ་ན། །ངན་སོང་འཁོར་བའི་སྡུག་བསྔལ་བཟོད་དཀའ་བས། །སྲོག་ལ་བབས་ཀྱང་སྡིག་ལ་འཛེམ་པར་མཛོད། །སེམས་ཅན་དམྱལ་བའི་གནས་ནའང་སྡུག་བསྔལ་ཚད་མེད་དོ། །ཡང་སོས་ཐིག་ནག་བསྡུས་འཇོམས་ངུ་འབོད་ངུ་འབོད་ཆེ། །ཚ་བ་རབ་ཚ་མནར་མེད་ཉི་འཚེ་ཉེ་འཁོར་སོགས། དེ་དག་ལྷུན་ཏུ་མི་སྐྱེའི་ཐབས་ལ་འབོད་པར་མཛོད། ཡི་དྭགས་སུལ་ནའང་ཤིན་ཏུ་སྡུག་བསྔལ་ཏེ། ཕྱི་ནང་སྒོས་ཁུར་སྒྲིབ་པ་ཅན་ལ་སོགས། །ཟས་སྐོམ་མི་བརྙེད་བརྙེད་ཀྱང་ཁྲོ་ཆུར་འགྱུར། །བསྲུང་བར་མཐོང་སོགས་དཔག་མེད་མཁྱེན་པར་མཛོད། །དུད་འགྲོའི་གནས་ནའང་ཤིན་ཏུ་སྡུག་བསྔལ་ཏེ། །གཅིག་གིས་གཅིག་གསོད་དབང་མེད་བཀོལ་སྤྱོད་བརྡེག །མཁལ་འགེལ་ཁྲག་ ༤༨༠༽བཙགས་བལ་བྲེག་ཤ་ཕྱིར་གསོད། །དེ་སོགས་སྡུག་བསྔལ་དཔག་མེད་མཁྱེན་པར་མཛོད། །གང་ཞིག་མི་རྣམས་ལ་ཡང་སྡུག་བསྔལ་ཚད་མེད་དེ། །སྐྱེ་རྒ་ན་འཆི་དགྲ་དང་ཕྲད་ཅིང་གཉེན་བྲལ་དང༌། །འདོད་པའི་དོན་མ་གྲུབ་དང་དབུལ་ཞིང་འཕོངས་པ་སོགས། །རྟག་ཏུ་སྡུག་བསྔལ་འདམ་དུ་བྱིང་བ་མཁྱེན་པར་མཛོད། །གང་ཞིག་ལྷ་མིན་གནས་ནའང་ཞེ་སྡང་དབང་གྱུར་པས། །དོན་མེད་འཐབ་བརྩོད་འཁྲུག་དང་གཅིག་གིས་གཅིག་གསོད་ཅིང༌། །རལ་གྲི་མདུང་ཐུང་ཤཀྟི་ལ་སོགས་འཕེན་བྱེད་པ། །འཐབ་བརྩོད་སྡུག་བསྔལ་ཤིན་ཏུ་ཆེ་བ་མཁྱེན་པར་མཛོད། །ལྷ་ཡུལ་རྣམས་སུ་སྐྱིད་པར་སྣང་ཡང་འགྱུར་བ་སྟེ། །ལུས་ཀྱི་མདོག་ཉམས་འོད་ཀྱང་ཡལ་ཞིང་གྲོགས་ཀྱིས་བོར། །ཕྱི་མའི་སྡུག་བསྔལ་སྐྱེ་གནས་ངན་མཐོང་སྙིང་མི་དགའ། །འཕོ་ལྟུང་སྡུག་བསྔལ་ཤིན་ཏུ་བཟོད་དཀའ་མཁྱེན་པར་མཛོད། །དེ་ལྟར་འཁོར་བ་མེ་འོབས་ ༤༨༡༽ལྟ་བུ་ལས། །རྟག་ཏུ་མི་ཐར་ཁྲི་མུན་ཆུད་བཞིན་དུ། །བདེ་བའི་གོ་སྐབས་ཅུང་ཟད་ཙམ་མེད་པ། །འདི་ལས་ད་ལྟ་ཉིད་དུ་བྲོས་རེ་རན། །འཁོར་བ་མི་བཟད་སྡུག་བསྔལ་རྒྱ་མཚོ་ཆེ་འདི་ལས། །ཐར་བྱེད་མི་ལུས་རིན་ཆེན་གྲུ་བོ་རྙེད་དུས་འདིར། །དེད་དཔོན་ཁ་ལོ་བསྒྱུར་བྱེད་ལྟ་བུའི་བླ་མ་ལ། །རབ་ཏུ་དད་དང་མོས་གུས་ལྡན་པས་རྟག་བསྟེན་མཛོད། །གང་ཞིག་ནོར་བུ་རིན་ཆེན་བརྙེད་ནས་སླར་འདོར་ལྟར། །འདྲེན་མཆོག་མཚན་ལྡན་བླ་མ་བརྙེད་ཀྱང་མི་བསྟེན་ཞིང༌། །དད་པའི་དམ་ཆོས་བདུད་རྩི་འཐུང་བར་མི་བྱེད་པའི། །མི་དེ་རབ་ཏུ་བླུན་རྨོངས་ལགས་པར་མཁྱེན་པར་མཛོད། །གང་ཞིག་རོ་མྱགས་འདམ་ཕྱིན་མི་གཙང་གོས་པ་བཞིན། །སྡིག་པའི་གྲོགས་དང་འགྲོགས་པས་རང་ཡང་ངན་པར་འགྱུར། །དམ་པ་རྣམས་དང་འགྲོགས་པས་རང་ཡང་དམ་པར་འགྱུར། །དེ་ཕྱིར་སྡིག་པའི་གྲོགས་

༤༨༢༽པོ་སྤང་ལ་འབད་འཚལ་ལོ། དེ་ལྟར་ལེགས་པར་བཤད་པ་ཅུང་ཟད་ཙམ་བརྗོད་ནས། །ད་ནི་སྔོན་བྱུང་གི་གཏམ་ལ་བརྟེན་ནས་བརྗོད་པར་བྱ་སྟེ། དེ་ཡང་འདི་ལྟར་སྔོན་བྱུང་བ་འདས་པའི་དུས་ན། འོ་སྐོལ་གྱི་སྟོན་པ་ཤཱཀྱའི་དབང་པོ་འདི་ཉིད་བྱང་ཆུབ་ཀྱི་སྤྱད་པ་སྤྱོད་པའི་ཚེ། རེ་ཞིག་ཅིག་ན་ཚོང་པ་རྣམས་ཀྱི་དེད་དཔོན་ཆེན་པོ་ཞིག་ཏུ་སྐྱེ་བ་ཡོངས་སུ་ཟིན་པར་གྱུར་ཏོ། ཞེས་གྲག་གོ། དེ་ནི་གཟུགས་བཟང་སྙིང་སྟོབས་ཆེ་བ་དང༌། །སྙིང་རྗེ་ཆེ་ཞིང་གཞན་དོན་རྟག་ཏུ་བརྩོན། །ངང་རྒྱུད་མཆོག་ཏུ་རིང་ཞིང་བརྩོན་འགྲུས་ལྡན། །གཞན་ཕན་བསྒྲུབ་ལ་རྟག་ཏུ་གཡེལ་མེད་གནས། དེ་ལྟ་བུའི་བྱང་ཆུབ་སེམས་དཔའ་དེ་རེ་ཞིག་ཅིག་ཚོང་པ་ལྔ་བརྒྱ་དག་དང་ལྷན་ཅིག་ཏུ་ཕྱིའི་རྒྱ་མཚོ་ཆེན་པོར་ནོར་བུ་རིན་པོ་ཆེ་ལེན་པའི་སླད་དུ། གྲུ་གཟིངས་ཀྱི་ཡོ་བྱད་དང་བཅས་ཏེ་གྲུ་བོ་ཆེའི་ནང་དུ་ ༤༨༣༽ཞུགས་པར་གྱུར་ཏོ། དེ་ནས་དེ་དག་རྒྱ་མཚོ་ཆེན་པོར་ཉིན་ཞག་མང་པོར་སོང་སོང་བ་ལས་རེ་ཞིག་ཅིག་ན་ཚོང་ཕྲུག་རྣམས་བསོད་ནམས་དམན་ཞིང་བདུད་ཀྱི་རིགས་གྱུར་པ་རྣམས་སྟོབས་ཆེ་བ་དང༌། དཀར་ཕྱོགས་ཀྱི་ལྷ་ཀླུ་རྣམས་མཐུ་ཆུང་ཞིང་རྩལ་མེད་པར་གྱུར་ནས། འདི་ལྟར་སྟེ། ནམ་མཁར་སྤྲིན་ནག་གི་ལྡིང་གུར་ནི་རབ་ཏུ་ཕུབས། ཉི་ཟླ་དང་སྐར་ཚོགས་ནི་སྣང་བ་མེད་པར་གྱུར་རོ། རླུང་ཆེན་པོའི་སྒྲ་ནི་རབ་ཏུ་ཆེས་ཆེར་སྒྲོགས། ཆར་གྱི་ཚོགས་དྲག་པོ་ནི་ཉིན་མཚན་བར་མེད་དུ་འབབ་ཅིང༌། གྲུ་བོ་ཆེ་དེ་ཉིད་ཀྱང་མི་འདོད་པའི་རླུང་གིས་རབ་ཏུ་བསྐྱོད་ནས་སྲིན་མོའི་གླིང་གི་ཕྱོགས་སུ་དཔག་ཆེན་གྱི་མདའ་འཕངས་པ་ལྟར་འཁྱེར་བར་གྱུར་ཏོ།། ཆུ་འཛིན་བུང་བ་ལྟ་བུར་རབ་གནག་ཅིང༌། །འབྲུག་སྟོང་ལྡིར་བ་ལྟ་བུའི་སྒྲ་ཡང་སྒྲོགས། །ཆར་གྱི་རྒྱུན་ཡང་རབ་ཏུ་ཆེར་འབབ་པས། །གྲུ་ཡང་འཆི་བདག་ཕྱོགས་སུ་ ༤༨༤༽ཉེ་བར་གྱུར། དེ་ལྟར་གྲུ་བོ་ཆེ་དེ་ཉིད་མི་འདོད་པའི་རླུང་གི་ཤུགས་ཆེན་པོས་བསྐྱོད་ནས།

སིང་ག་ལ་ཞེས་བྱ་བའི་སྲིན་མོའི་གླིང་ཞིག་ཏུ་ཕྱིན་པར་གྱུར་ཏེ། ཡུལ་དེར་གནས་པའི་སྲིན་མོ་རྣམས་ཀྱང་འཛམ་བུ་གླིང་གི་ཚོང་པ་འོངས་པ་མཐོང་སྟེ། ལྷ་མོ་ཤིན་ཏུ་མཛེས་པ་ལྟ་བུའི་གཟུགས་སུ་སྒྱུར་ནས་ཚོང་པ་རྣམས་ལམ་དུ་བསུས་ཤིང་ཟས་སྐོམ་གྱིས་ཚིམ་པར་བྱས་ཏེ། བདག་ཅག་གི་ཡུལ་འདིར་ཞུགས་ཤིག་ཅེས་སྨྲས་ནས། སྲིན་མོའི་དཔོན་དེས་ཚོང་དཔོན་འཁྲིད་ཅིང་གཞན་རྣམས་ཀྱིས་ཀྱང་ཚོང་པ་རེ་རེ་བཞིན་ཁྲིད་ནས་ལྷོ་ཕྱོགས་སུ་མ་བྱོན་ཞིག་ཟེར་ཏེ་སོ་སོར་ལྷན་ཅིག་གནས་པར་གྱུར་ཏོ། ཚོང་པ་རྣམས་ཀྱིས་ཀྱང་དེ་དག་སྲིན་མོ་ཡིན་པ་ནི་མ་རིག་ཅིང་གཟུགས་དང་ལོངས་སྤྱོད་སོགས་ལ་ཡིད་རབ་ཏུ་མགུ་བས་གནས་སོ། སྲིན་མོ་ལྷ་མོར་སྤྲུལ་པའི་གཡོ་རྒྱུ་ཡིས། །མི་ཤེས་རྨོངས་པའི་ཚོང་པ་བསླུས་པ་ལྟར། །མ་རིག་མུན་ ༤༨༥༽པར་འཁྱམས་པའི་འགྲོ་བ་ཀུན། །འདོད་ལ་ཆགས་པས་འཁོར་བར་ལྷུང་འདི་མཚུངས། དེ་ལྟར་ཚོང་པ་དེ་དག་སྲིན་མོ་རྣམས་དང་ལྷན་ཅིག་ཏུ་གནས་པ་ལས་རེ་ཤིག་གི་ཚེ། ཚོང་དཔོན་ཆུང་མ་དང་ལྷན་ཅིག་མལ་དུ་ཉལ་བའི་སྲིན་མོ་དེས་བླ་རྡོལ་དུ་འདི་སྐད་ཅེས་ཟེར་ཏེ། ཡུལ་འདི་ནི་སིང་ག་ལའི་གནས་ཡིན་ཞིང་བདག་ཅག་ནི་སྲིན་མོ་ཡིན་ཏེ་ཁྱོད་ཅག་གི་སྲོག་གཅོད་པར་བྱའོ། ཚུལ་དེས་ཁྱེད་ཅག་ཡིད་མ་ཆེས་ན། གྲོང་ཁྱེར་འདིའི་གཡས་ཀྱི་སྲང་ལ་གཞུད་པས་ལྷོ་ཕྱོགས་ན་ལྕགས་ཀྱི་ཁང་པ་རབ་ཏུ་འཕངས་མཐོ་བ་དཀར་ཁུང་དང་རྟ་བབ་མེད་པ་ཞིག་མཆིས་ཀྱི། དེའི་ནང་དུ་ཚོང་པ་ལྔ་བརྒྱ་ཙམ་བོར་བ་དག་ལ་ལ་གསོན་ཞིང༌། ལ་ལ་གུམ་སྟེ་མཆིས་པས་དེར་གཟིགས་ཤིག ཅེས་ཟེར་བས། ཚོང་དཔོན་གྱིས་སྲིན་མོ་དེས་མ་ཚོར་བར་ལྷོ་ཕྱོགས་དེར་ཕྱིན་ཏོ། ཕྱིན་ནས་ཀྱང་ལྕགས་ཀྱི་ ༤༨༦༽གྲོང་ཁྱེར་དེའི་ཁོ་ར་ཁོར་ཡུག་ཏུ་བཙལ་ཀྱང་སྒོ་མ་རྙེད་ནས། ལྕགས་ཁང་གི་ཕྱོགས་གཅིག་ན་ལྗོན་ཤིང་ཙམ་པ་ཀའི་སྡོང་པོ་ཆེན་པོ་ཞིག་འདུག་པ་ལ་འཛེགས་སྟེ་ནང་དུ་བལྟས་པ་ན། ལྕགས་ཁང་ནང་གི་ཚོང་པ་དེ་རྣམས་ཀྱིས་ཚོང་དཔོན་ལ་འདི་སྐད་ཅེས་ཟེར་རོ།

ཚོང་དཔོན་ཆེན་པོ་ཁྱེད་ཅག་མཁྱེན་པར་མཛོད་ཅིག ཡུལ་འདི་ནི་སྲིན་མོའི་གླིང་ཡིན་ཏེ་བདག་ཅག་རྣམས་གསོད་པའི་ཕྱིར་ལྕགས་ཁང་འདིར་སྩལ་བར་གྱུར་ཏོ། འཛམ་བུའི་གླིང་གི་ཚོང་པ་འོངས་པ་ཀུན་ཕྱི་མ་རྣམས་རྩེ་དགས་བྲིད་ཅིང་སྔ་མ་རྣམས་ལྕགས་ཁང་འདིར་བཅུག་ཏེ། ཉེན་རེ་བཞིན་དུ་མི་བརྒྱ་བརྒྱ་ཁྲིད་ནས་ཟོས་ཤིང་རུས་པ་རྣམས་ལྕགས་ཀྱི་གྲོང་འཁྱེར་འདིར་བོར་བར་བྱེད་དོ། བདག་ཅག་ནི་གོ་ཞིང་ཤེས་ཀྱང༌། ལྕགས་ཁང་ལ་འཛེག་ན་ནི་མཐོ་རུ་འགྲོ། བྲུས་ན་ནི་གཏིང་རིང་དུ་འགྲོ་བས་ཐབས་མེད། ཁྱེད་ཅག་རྣམས་བྲོས་ན་ཐར་བས་བྲོས་པ་དྲག ཅེས་ཟེར་བས་ ༤༨༧༽ཚོང་དཔོན་ཡང་དེ་མ་ཐག་མྱུར་དུ་རང་གི་ཁྱིམ་དུ་ལོག་ནས་ཉལ་བས། སྲིན་མོ་ནི་གཉིད་ཀྱིས་ནོན་པའི་སྟོབས་ཀྱིས་མ་ཚོར་བ་ལ་སླར་ཡང་ཚོང་དཔོན་གྱིས་དྲིས་པ། འོ་ན་འདི་ལས་ཐབས་གང་གི་ཐར་སྨྲས་པས་འདི་སྐད་ཟེར་རོ། རྟ་མཆོག་རྒྱལ་པོ་བ་ལ་ཧ། །ཞེས་བྱ་སྙིང་རྗེ་ལྡན་པ་ཞིག །བཅོ་ལྔའི་ཟླ་བ་ཉ་ཚེ་འབྱུང༌། །རྟ་མཆོག་ཅང་ཤེས་ལྡན་པ་དེས། །བྷེ་ཏ་ཐམས་ཅད་ཅེས་བྱ་བའི། །སྨན་གྱི་རྒྱལ་པོ་ཟོས་ནས་ནི། །བཻ་ཌཱུརྻ་ཡི་བྱེ་མའི་དབུས། །ཆུ་རྒྱུན་ཡན་ལག་བརྒྱད་པ་འཐུངས། །གསེར་གྱི་བྱེ་མའི་ངོས་ལ་ནི། །ཕན་ཚུན་འགྲོ་ཞིང་སྤྲུག་པར་བྱེད། །འདི་ནས་ཕ་རོལ་སུ་འགྲོ་ཞེས། །མི་ཡི་སྐད་དུ་སྨྲ་ཞིང་འཚེར། །དེ་ཚེ་ཁྱེད་ཀྱི་རྟ་རྒྱལ་མཆོག །བདག་ནི་ཕ་རོལ་སྒྲོལ་ཞེས་སྨྲོས། དེ་སྐད་ཅེས་ཟེར་ཏེ་མལ་དུ་ལྷན་ཅིག་ཉལ་ ༤༨༨༽བཞིན་པ་ནས་སྲིན་མོ་གཉིད་ལས་སད་དེ། ཚོང་དཔོན་ཆེན་པོ་གར་བྱོན། ལུས་གྲང་བ་ལ་ཟེར་བས། བདག་གི་གྲོང་ཁྱེར་གྱི་ཕྱི་རོལ་དུ་ལུས་འཆག་ཏུ་ཕྱིན་ནོ་ཞེས་སྨྲས་སོ། སྲིན་མོས་རང་ཚང་རང་གིས་སྨྲས་པ་བཞིན། །འཁོར་བའི་མཐར་ཐུག་རང་ཚང་རང་གི་སྟོན། །འགྲོགས་ཤིང་ངན་པའི་འཁོར་བའི་འདམ་འདི་ལས། །ད་ལྟ་ཉིད་དུ་ཐར་བའི་ཐབས་འཚལ་ལོ། །དེ་ལྟར་ཚོང་དཔོན་གྱི་ཚུལ་དེ་དག་རིག་ནས།

ཉི་མ་ཤར་བ་དང་ཚོང་ཕྲུག་རྣམས་ཁྲིད་ནས་དབེན་པ་ཞིག་ཏུ་སོང་སྟེ། ཡུལ་འདི་སྐྱིད་དམ། མཆིས་འགྲང་དང་འཛའ་འམ། ཞེས་སོགས་དྲིས་པས་ཚོང་པ་དེ་དག་ལས་ཁ་ཅིག་ནི། ཤིན་ཏུ་སྐྱིད་དེ་བདག་ལ་ལྷའི་ཟས་སྐོམ་དང་གོས་དང་རྒྱན་དག་གི་ཡིད་རབ་ཏུ་འཛིན་ཏོ། ཞེས་ཟེར། ཁ་ཅིག་ནི་བདག་གིས་མཆིས་འབྲང་གཟུགས་དང་ལང་ཚོ་ལྡན་པས་དགའོ། ཞེས་ཟེར། ཁ་ཅིག་ནི་བྲེལ་བ་ཅི་ཡང་ ༤༨༩༽མེད་པས་སྐྱིད་དོ། ཞེས་སོགས་ཟེར་བ་ལ། ཚོང་དཔོན་གྱིས་བདག་ཅག་རྣམས་སྲིན་མོའི་ཡུལ་ལ་ཆགས་པར་མི་རིགས་སོ། ཞེས་སྨྲས་པས་དེ་རྣམས་ན་རེ། སྲིན་མོ་ཡིན་པའི་རྒྱུ་ཅི་ཞེས་ཟེར་བ་ལ། ཚོང་དཔོན་གྱི་སྲིན་མོས་བླ་བརྟོལ་དང་ལྕགས་ཁང་གི་ལོ་རྒྱུས་སོགས་ཞིབ་ཏུ་བརྗོད་པས། དེ་དག་རབ་ཏུ་བསྐྲག་ཏེ། འོ་ན་ཐབས་གང་གིས་ཐར་ཞེས་ཟེར་བ་ལས། ཚོང་དཔོན་གྱིས། ཚེས་བཅོ་ལྔའི་ནུབ་མོ་སོ་སོ་ནས་བུད་མེད་རྣམས་ཆང་གིས་མྱོས་པར་གྱིས་ལ་ས་ཕྱོགས་འདི་ཞེས་བྱ་བར་འདུས་ཤིག ཅེས་བསྒོས་ནས་སོ་སོར་གྱེས་སོ། དེ་ནས་ཚེས་བཅོ་ལྔའི་ནུབ་མོ་ཟླ་བ་ཤར་བའི་ཚེ་རྟ་མཆོག་དེ་ཟླ་བའི་ཟེར་དང་འགྲོགས་ནས་གླིང་དེར་འོངས་ཏེ། སྨན་མཆོག་བྷེ་ཏ་ཟོས་ཤིང་བདུད་རྩིའི་ཆུ་འཐུང་སྟེ་འགྲོ་ཞིང་སྤྲུག་ནས། ཚོང་ཕྲུག་སྲིན་མོའི་གླིང་དུ་ཆུད་པ་སུ་ཡོད། ཐམས་ཅད་ངའི་ ༤༩༠༽རྒྱབ་ཏུ་བཞོན་ཅིག་ཅེས་ཟེར་བ་ལ་ཚོང་པ་ཐམས་ཅད་ཀྱིས་མགྲིན་གཅིག་ཏུ་བདག་ཅག་རྣམས་སྒྲོལ་བར་མཛད་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། རྟ་མཆོག་དེས་ཁྱེད་ཅག་སུས་ཀྱང་གླིང་འདིར་ཆགས་པར་མ་བྱེད་ཅིག ཅེས་བསྒོས་ནས། ཐོག་མར་ཚོང་དཔོན་གྱིས་ཞོན་ཞིང་རིམ་པ་ལྟར་གཞན་ཐམས་ཅད་ཀྱིས་ཞོན་ཏེ་བར་སྣང་ལ་བྱ་བཞིན་འཕུར་ནས་འཛམ་བུ་གླིང་གི་ཕྱོགས་སུ་ཆས་པར་གྱུར་ཏོ། དེའི་ཚེ་སྲིན་མོ་རྣམས་ཀྱིས་ཚོར་ཏེ་ཕྱི་བཞིན་དུ་བསྙེགས་ནས་ཆོ་ངེ་དང་སྨྲེ་ངག་ཆེན་པོས། བདག་ཅག་བོར་བར་ཕོད་དམ། བུ་དང་བུ་མོ་རྣམས་ལ་སྙིང་ཡང་མི་རྗེའམ། ཞེས་བོས་པའི་ཚེ་ཚོང་ཕྲུག་ཀུན་དེ་དག་ལ་ཆགས་ནས་ཕྱི་མིག་བལྟས་པ་ཙམ་གྱིས་ས་ལ་ལྷུང་ཞིང་འགྱེལ་བ་ལས།

བུད་མེད་དེ་དག་སྔར་གྱི་སྤྲུལ་པ་བཤིག་སྟེ་སྲིན་མོའི་གཟུགས་མི་སྡུག་ཅིང་རབ་ཏུ་འཇིགས་པ་སྟེ། རལ་པ་གཟིངས་ཤིང་མིག་དམར་བ། མཆེ་བའི་དབལ་ ༤༩༡༽རབ་ཏུ་རྣོ་བ། རྔམ་པའི་གཟུངས་སུ་བསྒྱུར་ནས་ཚོང་པ་དེ་དག་གི་ཤ་ཟོས་ཤིང་ཁྲག་འཐུངས་ཏེ་སྐད་ཅིག་ལ་མ་ལུས་པར་བྱས་སོ། ཚོང་དཔོན་ནི་དེ་རྣམས་ལ་མ་ཆགས་ཤིང་ཕྱི་མིག་མ་བལྟས་པས་བདེ་བླག་ཏུ་འཛམ་བུའི་གླིང་དུ་ཐར་བར་གྱུར་ཏོ། ཚོང་དཔོན་གྱི་ཕ་མ་ཡང་མྱ་ངན་གྱིས་མིག་ལྡོངས་བར་གྱུར་ཏེ་གནས་པ་ལས། ཚོང་དཔོན་ཉིད་སླར་ལོག་པས་མིག་ཀྱང་སོར་ཆུད་ཅིང་རབ་ཏུ་དགའ་བར་གྱུར་ཏོ། རྟ་མཆོག་ནི་ཚོང་དཔོན་ལ་ལན་གསུམ་སྐོར་བ་བྱས་ནས་དེ་ཉིད་དུ་མི་སྣང་བར་གྱུར་ཏོ།། སྲིན་མོ་ལྷ་མོ་མཛེས་མར་བསྒྱུར་བ་ཡིས། །གཟུགས་དང་ལང་ཚོ་ཟས་སྐོམ་རྒྱན་སོགས་ཀྱིས། །བསླུས་པ་ཇི་བཞིན་ཚེ་འདིའི་འདོད་ཡོན་གྱིས། །བྲིད་ཅིང་ངན་འགྲོའི་གཡང་སར་བསྐྱུར་དེ་མཚུངས། །སྲིན་མོའི་གཟུགས་ལ་ཆགས་པས་ཐ་མར་ནི། །རང་གཟུགས་བསྟན་ནས་ཟ་བ་ཇི་བཞིན་དུ། ༤༩༢༽ །འདུས་བྱས་ཆོས་རྣམས་དང་པོར་མཛེས་ན་ཡང༌། །མཐའ་མར་མི་རྟག་འཇིག་པའི་ཆོས་ཅན་ནོ། །ཇི་ལྟར་སྲིན་མོས་ཚོང་དཔོན་བསླུས་གྱུར་ཀྱང༌། །གཉིད་ཀྱི་ནོན་པས་རང་ཚང་བརྟོལ་བ་ལྟར། །འདུས་བྱས་སྣང་བའི་འདོད་ཡོན་འདི་དག་ཀྱང༌། །ལེགས་པར་བརྟགས་ན་རང་མཚང་སྟོན་པ་འགྱུར། །ཇི་ལྟར་ལྕགས་ཁང་བོར་བའི་ཚོང་པ་དེ་དག་གིས། །སྲིན་མོའི་ངང་ཚུལ་མངོན་སུམ་བསྟན་པ་ཇི་བཞིན་དུ། །བར་དོ་གཤིན་རྗེའི་རྒྱལ་པོའི་དྲུང་དུ་བཏུགས་ཙ་ན། །བརྫུན་གྱིས་མི་སྐྱོབ་ལས་འབྲས་དགེ་སྡིག་གསལ་བར་སྟོན། །གང་ཞིག་དེ་རྣམས་ལྕགས་ཁང་བསྲ་བ་དེར་ཆུད་ནས། །གོ་ཞིང་ཤེས་ཀྱང་ཐབས་མེད་པ་ཤེས་ཇི་བཞིན་དུ། །དམྱལ་བའི་ལྕགས་ཁང་འབར་བ་དག་ཏུ་ཚུད་གྱུར་ཚེ། །དགེ་སྡིག་ཤེས་ཀྱང་ཐབས་མེད་གྱུར་འདི་མཁྱེན་པར་མཛོད། །གང་ཞིག་སྲིན་ཡུལ་འཁྱམས་རྣམས་འདོན་པའི་ཕྱིར། །རྟ་མཆོག་རྒྱལ་པོའི་བརྩོན་

༤༩༣༽པ་ཇི་བཞིན་དུ། །སངས་རྒྱས་བྱང་སེམས་ཐུགས་རྗེ་ལྡན་པ་རྣམས། །འགྲོ་ཀུན་སྡུག་བསྔལ་འདི་ལས་ཐར་བར་མཛད། །འོན་ཀྱང་བཅོ་ལྔ་ཟླ་བ་ཉ་ཚེས་མ་གཏོགས་པ། །དུས་རྣམས་ཀུན་ཏུ་རྟ་རྒྱལ་སྟོབས་ཆེན་མི་སྣང་ལྟར། །སྔོན་གྱི་སྨོན་ལམ་དད་པར་ལྡན་པ་མ་ལགས་པ། །ཀུན་གྱི་བཤེས་གཉེན་དམ་པ་རྙེད་དཀའ་མཁྱེན་པར་མཛོད། །གང་ཞིག་གཏིང་ཟབ་རྒྱ་མཚོ་ཆེན་པོས་བར་ཆོད་པའི། །རབ་ཏུ་ཉམས་དང་སྲིན་མོ་གནས་པའི་ཡུལ་དེ་ལས། །སྟོབས་ཆེན་ཐུགས་རྗེ་རྫུ་འཕྲུལ་ལྡན་པ་དེ་ཉིད་ཀྱིས། །སྐད་ཅིག་ཉིད་ལ་འཛམ་བུའི་གླིང་དུ་བསྒྲོད་མཛད་ལྟར། །བཤེས་གཉེན་ཐུགས་རྗེ་ལྡན་པ་དེ་ཉིད་ལ་བརྟེན་ནས། །མྱུར་དུ་རྒྱལ་བའི་གོ་འཕང་བསྒྲོད་ལ་བརྩོན་པར་མཛོད། །གང་ཞིག་རྟ་མཆོག་སྙིང་རྗེར་ལྡན་པས་ཕན་པར་སྨྲས་པ་ལ། །མི་ཉན་སྲིན་མོར་ཆགས་པས་ལྷུང་སྟེ་ཕུང་ ༤༩༤༽བ་ཇི་བཞིན་དུ། །དགེ་བའི་བཤེས་གཉེན་མཆོག་གིས་ལེགས་པར་གསུངས་ལ་མི་ཉན་པར། །ཚེ་འདིའི་ཁེ་གྲགས་འདུ་འཛི་ཆོས་བརྒྱད་ཕྱོགས་ལ་ཕྱིར་བལྟས་རྣམས། །འཁོར་བ་་མེ་འོབས་ནང་དུ་ལྷུང་སྟེ་ཐར་མེད་མཁྱེན་པར་མཛོད། །གང་ཞིག་བློ་གྲོས་རབ་ཏུ་བརྟན་ཞིང་མི་གཡོ་བ། །རྟ་རྒྱལ་ཚིག་ལ་ལེགས་པར་ཉན་པའི་ཚོང་དཔོན་དེས། །འཛམ་བུའི་གླིང་དུ་བདེ་བར་ཐར་ཞིང་ཕ་མ་ཡི། །མྱ་ངན་བསལ་ཞིང་ཉེ་དུ་དགའ་བར་བྱས་པ་བཞིན། །བཤེས་གཉེན་བསྟེན་ཞིང་གསུང་བཞིན་བསྒྲུབས་ནས་རང་དོན་མཆོག །ལེགས་པར་གྲུབ་ཅིང་ཕ་མར་གྱུར་པའི་སེམས་ཅན་ཀུན། །བྱང་ཆུབ་གོ་འཕང་མཆོག་ལ་འགོད་པར་བྱ་བར་རིགས།། དེ་ལྟར་སྔོན་བྱུང་གི་གཏམ་ལ་བརྟེན་ནས་ལེགས་པར་བཤད་པ་མེ་ཏོག་གི་འཕྲེང་བ་ལྟ་བུ་བྱིན་པའི་དགེ་བས། བདག་སོགས་ནམ་མཁའ་དང་མཉམ་པའི་སེམས་ཅན་ཐམས་ཅད་ཚེ་འདིའི་ཁེ་གྲགས་འདུ་འཛི་ཆོས་བརྒྱད་ཀྱི་དབང་དུ་མ་ ༤༩༥༽སོང་བར། ཡང་དག་ངེས་དོན་སྙིང་པོའི་དོན་མཆོག་ལེགས་པར་གྲུབ་སྟེ། ཀུན་ཏུ་བཟང་པོའི་ཞིང་དུ་གདོད་མའི་མགོན་པོར་སངས་རྒྱས་པར་གྱུར་ཅིག།

དེ་ལྟར་ལེགས་བཤད་ཅུང་ཟད་བཀོད་པའི་མཐུས། །འགྲོ་ཀུན་སྡུག་བསྔལ་འདམ་ལས་ཐར་ནས་ཀྱང༌། །འོད་གསལ་མཆོག་གི་གོ་འཕང་རབ་ཐོབ་ནས། །ཀུན་བཟང་ཀློང་དུ་དབྱེར་མེད་གཅིག་པར་ཤོག །བདག་ཀྱང་འདི་ནས་ཚེ་རབས་ཐམས་ཅད་དུ། །རང་དོན་རྫོགས་སངས་རྒྱས་ཤིང་གཞན་དོན་དུ། །གཡེལ་བ་མེད་པར་རྟག་ཏུ་བརྩོན་པ་དང༌། །ཤཱཀྱའི་ཏོག་བཞིན་ཀུན་གྱི་མགོན་དུ་ཤོག གཞན་ཡང་དགེ་དེས་གནས་སྐབས་དུས་ཀུན་ཏུ། །ཆར་ཆུ་དུས་བབས་ལོ་ཕྱུགས་ལེགས་པ་དང༌། །ནད་ཡམས་འཁྲུག་བརྩོད་མུ་གེ་ཀུན་ཞི་ནས། །རྫོགས་ལྡན་དུས་བཞིན་བདེ་སྐྱིད་ཕུན་ཚོགས་ཤོག །དེ་ལྟར་བདག་གི་ལྷག་བསམ་རྣམ་དག་དང༌། །སངས་རྒྱས་བྱང་སེམས་ཐུགས་རྗེའི་ ༤༩༦༽མཐུ་སྟོབས་ཀྱིས། །སྨོན་ལམ་གང་བཏབ་མྱུར་དུ་འགྲུབ་པ་དང༌། །མཐར་ཐུག་རྣམ་མཁྱེན་གོ་འཕང་མྱུར་ཐོབ་ཤོག། འདི་ལྟར་ཚིག་དང་དོན་གྱི་འཕྲེང་བ་དམ་པ་ལེགས་པར་བཀོད་པ། སྐྱེ་བོ་ཆོས་དང་མི་ལྡན་པ་རྣམས་ཆོས་ལ་བསྐུལ་བའི་ཕྱིར་དུ་ཅུང་ཟད་བརྗོད་པ་འདི་ནི། དུས་མཐར་སྐྱེས་པའི་སྙོམས་ལས་པ་མཁས་མཆོག་པདྨ་ལེགས་གྲུབ་རྩལ་ཞེས་བགྱི་བས་རང་ལོ་བཅུ་བཞི་པའི་ཚེ། ཀོང་ཡུལ་སྨད་ཀྱི་སའི་ཐིག་ལེ། སྨན་བཙུན་ཆེན་མོ་རྡོ་རྗེ་བོད་ཁམས་སྐྱོང་གི་ཕོ་བྲང་ཀོང་མཁར་ལྡེམ་པ་དང་ཉེ་བའི་ཡུལ། ཆུ་བོ་ལོ་ཧི་ཏའི་གཡས་འགྲམ། གྲུབ་པའི་དབང་ཕྱུག་ཐང་སྟོང་རྒྱལ་པོས་བྱིན་གྱིས་བརླབས་པའི་གནས་སུ་དམ་ཆོས་ཀྱི་ཕྲུ་མར་སྦྱར་བ་དགེ་ལེགས་འཕེལ།། 

འདི་འདྲའི་ཚིག་ཚོགས་འགའ་དང་འགའ་བྲིས་ཀྱང༌། །གཅིག་ཏུ་བདག་འདྲ་ཐོས་བསམ་སྦྱངས་སྟོབས་མེད། །གཉིས་སུ་ན་གཞོན་བྱིས་པའི་རྩོམ་པར་སོང༌། །གང་ལ་ནོར་འཁྲུལ་མང་མོད་མཁས་ལ་བཤགས།། མངྒ་ལཾ། །
