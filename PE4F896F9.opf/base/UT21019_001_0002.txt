༢༠༡༽༄༅། རྫུན་རིས་ཆུ་ཟླའི་ཟློས་གར་ཞེས་བྱས་བ་རྨི་ལམ་ཉིང་འཁྲུལ་གྱི་བཀོད་པ་བརྗོད་པའི་ཡི་གེ་སྣ་ཚོགས་རང་གྲོལ་གྱི་རྣམ་ཐར་ཁ་སྐོང་ལགས། ༢༠༢༽ན་མཿཨུཏྟཱ་རེ་བྷྱཿ བདག་སྙིགས་དུས་ཀྱི་བཙུན་པ་སྙོམས་ལས་པ། །གནས་ངོ་མཚར་དཔལ་གྱི་རྒོད་ཚང་ཕུག །ཕ་བླ་མ་མཁའ་འགྲོས་བྱིན་བརླབས་པར། །དུས་རེ་ཤིག་དབེན་པ་བསྟན་པའི་ཚེ། །ཡོངས་རྟ་ལོ་རྒྱལ་ཟླའི་ཚེས་བཅུ་དགུ། །དགུང་ཐོ་རངས་གཉིད་ཀྱི་འཁྲུལ་ཉམས་སུ། གཟུགས་མི་མཐོང་བུད་མེད་སྐད་སྙན་པ། །གྲྭ་བསྡེབས་པའི་གླུ་དབྱངས་འདིར་ལྟར་ཐོས། །གནས་འོག་མིན་ཆོས་ཀྱི་ཕོ་བྲང་ནས། །ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས་བཞུགས། །སྟོན་པ་ཆོས་ཀྱི་སྐུ་ལ་ཕྱག་འཚལ་ལོ། །ཤར་རི་བོ་ཏ་ལའི་གནས་མཆོག་ན། །འཕགས་པ་ཐུགས་རྗེ་ཆེན་པོ་ ༢༠༣༽བཞུགས། །ལོངས་སྐུ་སྤྱན་རས་གཟིགས་ལ་ཕྱག་འཚལ་ལོ། །གནས་རྔ་ཡབ་དཔལ་རིའི་ཕོ་བྲང་ནས། །ཨོ་རྒྱན་པདྨ་འབྱུང་གནས་བཞུགས། །བླ་མ་སྤྲུལ་པའི་སྐུ་ལ་ཕྱག་འཚལ་ལོ། །གནས་མཆོག་གཙང་པོ་མགོ་དགུ་ན། །བཞུགས་ས་ལེགས་ལ་བཞུགས་སྙིང་སྤྲོ། །བཞུགས་ས་ཤར་ལ་དགོན་པ་ནུབ། །དྲིན་ཅན་བླ་མ་མཆོག་ལ་ཕྱག་འཚལ་ལོ། །ཞེས་དེ་སྐད་ཟེར་འཕྲོར་གཉིད་ལས་སད། །རང་རླུང་སེམས་འཁྲུལ་སྣང་འཆར་སྒོ་ལ། །དོན་བདེན་པའི་སྙིང་པོ་མ་མཆིས་ནའང༌། །ཚིག་སྐུ་གསུམ་བླ་མར་ཕྱག་འཚལ་བ། །ཡིད་མཚར་སྙམ་ཡི་གེར་བྲིས་པ་ལགས།། ༈ ༢༠༤༽ཡང་ལན་ཅིག་འཁྲུལ་པའི་རྨི་ལམ་ན། །ན་མཉམ་པའི་བཙུན་མ་ཆས་བཟང་ལྔ། །མི་ཁོ་བོར་གུས་པའི་རྣམ་འགྱུར་བཅས། །གཏམ་གླེང་མོ་མང་ཞིག་བགྱིས་པའི་མཐར། །ཟབ་དོན་གྱི་ཆོས་འབྲེལ་དགོས་སོ་སྐད། །བདག་དམ་ཆོས་འཆད་དོ་སྙམ་པའི་ཚེ། །ཆས་ནོར་བུའི་རིགས་བྱེད་ཡ་མཚན་ཅན། །སྣ་མང་པོ་སོ་སོས་བདག་གི་ལག །དུས་གཅིག་ཏུ་གཏད་པ་རང་ཉིད་ཀྱིས། །གྲལ་རིམ་པ་བཞིན་དུ་བླང་སྙམ་ཆེ། །ཁོང་དབུས་མ་དེ་ཡི་ལག་ཉིད་ནས། །སྣོད་ཐོད་པ་ཡིད་འོང་གཅིག་གི་ནང༌། །ནས་དཀར་མོ་འབྲུ་ཡིས་བཀང་བའི་སྟེང༌། །ཏིང་འདྲ་བའི་རིན་ཆེན་སྣོད་གཅིག་ཏུ། །ཤེལ་དཀར་པོ་དྲི་མེད་སྒོ་ང་ཙམ། །སྣོད་གཅིག་ནི་ཛ་གད་བདུད་རྩིས་བཀང༌། །གཞན་ངོ་མཚར་འགའ་ཡང་འདུག་སྙམ་བྱེད། །ཀུན་ལྷན་ཅིག་བདག་གིས་བླངས་པའི་མོད། །གཉིས་སད་ཅིང་སྤྲོ་བའི་ཡིད་སྣང་ནི། །ཡུན་རིང་ཞིག་བྱུང་

༢༡༠༽བདག་ལ་སྨྲ་བར་གྱིས། །དེ་སྐད་བྱས་པས་ཁོ་ན་རེ། །བསྐལ་པའི་འཕོ་འགྱུར་འདི་བཞིན་ནི། །འདུས་བྱས་ཆོས་ཉིད་ངང་ཚུལ་ཡིན། །བསྟན་པའི་དར་རྒུད་བསྟན་འཛིན་རྣམས། །བྱ་བྱེད་ཉེས་པའི་འབྲས་བུ་ཉིད། །ཤཱཀྱའི་བསྟན་པ་བསྟན་འཛིན་གྱི། །ཆ་ལུགས་ཉིད་ཀྱིས་བསྣུབས་འགྱུར་བ། །ལུང་དུ་བསྟན་པ་མ་ཐོས་སམ། །བདག་ཅག་བསྟན་པ་བསྲུང་རྣམས་ཀྱང༌། །བསྲུང་ཡུལ་མེད་ཚེ་བསྲུང་ཐབས་བྲལ། །ད་དུང་རིམ་གྱིས་ངན་ངན་དུ། །འགྲོ་འགྱུར་སེམས་ཅན་ལས་དབང་ཡིན། །ཁྱོད་ཀྱང་དམ་ཆོས་འཆད་ཉན་གྱི། །གཞན་ཕན་བསམ་པ་བཟང་ན་ཡང༌། །ཁྱི་ལ་རྩྭ་ཡི་སྦྱིན་པའམ། །འོན་པའི་ཁྲོད་དུ་གླུ་ལེན་འདྲ། །ཚེ་ཡི་ཚད་ལ་འཁྱོལ་བ་ཡང༌། །དམ་སེལ་གྲིབ་ཀྱིས་དཀའ་བར་མཆིས། །སྐུ་ཚེའི་རྟེན་འབྲེལ་བསྒྲིག་འདོད་ན། །གནས་བཟང་དུས་ཚེས་བཟང་པོ་ལ། །རྡོ་རྗེ་རྒྱ་གྲམ་ཟུར་རྣམས་སུ། །ནོར་བུས་སྤྲས་པའི་གདན་བཤམས་ནས། ༢༡༡༽།པཀྵི་ཝརྞ་ཀུ་མ་རིས། །ག་ཎཱ་ཙཀྲའི་པཱུ་ཛར་བཅས། །གསོལ་འདེབས་སྨོན་ལམ་མང་བགྱིས་ན། །ཚེ་ལ་ཕན་པར་འགྱུར་ན་ཡང༌། །སྙིགས་དུས་རྨུ་རྒོད་ལོག་ལྟ་ཡི། །དུག་གིས་སྨྱོས་པའི་སྐྱེ་བོ་རྣམས། །ཕྱིས་ཕྱིར་ལོག་པར་ལྟ་བ་འཕེལ། །དེ་ཡིས་ནམ་ཡང་འགྲུབ་པར་དཀའ། །དེ་སྐད་ཟེར་བའི་མོད་ཉིད་ལ། །གཉིད་ལས་སད་པར་གྱུར་པ་ན། །སེམས་ཀྱི་ཆོ་འཕྲུལ་སེམས་ཉིད་ཀྱིས། །མཐོང་བའི་འཁྲུལ་པའི་རྫུན་རིས་འདི། །དེ་དེ་བཞིན་ནོ་བདེན་ནོ་ཞེས། །རང་གི་རང་ལ་ཡི་རངས་སོ།། ༈ ཡང་ནི་དཔལ་གྱི་རྣམ་ཐོས་རི། །འཕོ་འཆུགས་གྱུར་དང་དུས་མཚུངས་པར། །ནུབ་གཅིག་ཉིང་འཁྲུལ་རྨི་ལམ་དུ། །ཀོང་པོའི་ཆ་བྱད་མི་དཔོན་གཟུགས། །མགུལ་དུ་རྡོ་རྗེ་རྩེ་ལྔ་བཏགས། །བག་དྲོ་སྣུམ་བརྗིད་ཆགས་པ་ཞིག །འཁོར་གཡོག་གཉིས་བཅས་བདག་དྲུང་འོངས། །ཕྱག་བཙལ་གུས་པའི་རྣམ་འགྱུར་བྱེད། །དེ་ལ་བདག་གིས་གཏམ་དྲིས་པ། །ཡོན་ ༢༡༢༽བདག་ཁྱེད་ཅག་གང་ཡིན་སྨྲས། །ང་ངོ་ཁྱེད་ཀྱིས་མི་ཤེས་སམ། །དགེ་བསྙེན་བཛྲ་སཱ་དྷུ་ཡིན། །བདག་ཅག་རྟེན་ཡུལ་མ་མཆིས་པས། །ཁྱེད་ཀྱི་ཉེ་སྐོར་འདི་ཤེད་ཅིག

ཁྱད་པར་དགོངས་འདུས་ལ་སོགས་པ། །ཟེར་འཕྲོར་བདག་ནི་གཉིད་སད་དེ། །ཁོ་ཡི་ཚིག་མཇུག་རྫོགས་མ་བྱུང༌། །རང་སེམས་རྩ་རླུང་འཁྲུགས་པ་ཡི། །རྣམ་རྟོག་བག་ཆགས་ཉིང་འཁྲུལ་ལ། བདེན་པའི་སྙིང་པོ་མ་མཆིས་ཀྱང༌། །དངོས་ཀྱི་ཚུལ་དང་རྨི་ལམ་འདི། །རྟགས་མཐུན་སྣང་ཕྱིར་མཚར་སྙམ་ནས། །ཡི་གེར་བྲིས་འདི་ཟོག་བརྫུན་དུ། །མཐོང་དོགས་སྦས་ན་ལེགས་སྙམ་ལགས། ༈ ཀུན་རྫོབ་སྣང་ངོར་སྲུང་མ་འདི་དག་ཀྱང༌། །གསོལ་མཆོད་རྟེན་ན་འགོ་བར་འགྱུར་སྲིད་དེ། །སྐྱབས་གནས་སྙིང་པོར་དཀོན་མཆོག་བླ་མ་ཆོག འཁྲུལ་གཟུགས་ལྷ་འདྲེའི་གྲོགས་ཀྱིས་ ༢༡༥༽ཅི་ཞིག་བྱ། །མ་བཅོས་ལྷུག་པའི་ངང་དུ་བསྐྱུར་ནས་བསྡད། །རྟག་ཏུ་གསོལ་མཆོད་བྱས་ཀྱང་མི་མགུ་ལ། །ནུབ་གཅིག་ཆག་ནའང་འཁང་ཞིང་འཁུ་བ་ཡི། །སྲུང་མ་དེ་འདྲར་རེ་བློ་བདག་ལ་མེད། །སྐྱིད་སྡུག་ཐམས་ཅད་བླ་མའི་དྲིན་དུ་ཕཊཿ ཨེ་མ། སྐྱབས་གནས་བླ་མ་རང་རིག་དབྱེར་མེད་ལ། །འདུ་འབྲལ་མེད་པར་རྟོགས་པས་ཕྱག་བྱས་ནས། །ཡང་ཅིག་སྙིགས་མཐའི་བན་ཆུང་སྙོམས་ལས་པ། །སྣ་ཚོགས་རང་གྲོལ་རྒན་པོ་འབྲུག་གི་ལོར། ཆུ་སྟོད་ཟླ་བའི་ཉེར་བདུན་མཚན་མོའི་གུང༌། །གཏི་མུག་གཉིད་དུ་ཉལ་བའི་འཁྲུལ་སྣང་ལ། །རྨི་ལམ་བར་དོའི་འཆར་ཚུལ་བཤད་ཀྱི་ཉོན། །ནུབ་ཏུ་ཁ་བལྟའི་ལུང་པ་གཅོང་རོང་ཟབ། །འཇིགས་རུང་ཆུ་བོ་རྦ་ཀློང་འཁྲུག་པའི་འགྲམ། །རྡོ་དང་ཚེར་མས་གཏམས་པའི་ལམ་འཕྲང་དུ། །གཤེད་མའི་སྐྱེས་བུ་མི་ལུས་སྟག་མགོ་ཅན། །མི་ཚད་ཉིས་ ༢༡༦༽འགྱུར་ཙམ་གྱི་གཟུགས་ལྡན་པ། །དེ་དང་འདྲ་ལ་གཤིན་རྗེའམ་ཁྲོ་བོའི་གདོང༌། །རྔམས་པ་གཅིག་དང་བཅས་པས་བདག་འཁྲིད་ནས། །ཉི་ནུབ་ས་སྲོས་ནུབ་ཏུ་རྒྱུགས་ཅིང་དེད། །གྲོགས་ནི་ར་ལུག་བ་ལང་འགའ་ཞིག་སྣང༌། །རླུང་ནག་ཐལ་འཚུབ་གནམ་ས་གཡོ་བ་ཙམ། །རྡོ་ཡི་ཆར་པ་སྒོང་དང་ཁུ་ཚུར་དང༌། །ཁམ་བུ་ཙམ་སོགས་སེར་དྲག་འབབ་ལྟར་བརྡུང༌། །ཆུ་གླུང་སྒྲ་ཡི་སེབ་ནས་ངར་སྐད་ཀྱིས། །ཆོས་མེད་བན་བཙུན་དཀོར་ཟན་བླ་གཟུགས་ཚོ། །སྐྱིད་པོ་ཡོད་མིན་འདི་རུ་བལྟའོ་ཟེར། །བར་སྐབས་གཅིག་ན་ཆུ་ཆེན་འགྲམ་འདི་རུ། །ཀ་པཱ་ལ་འདྲའི་མཚོ་ཆེན་ཁམ་ནག་འཁྲུག

།སེམས་བསྐྱེད་སྨོན་ལམ་སྙིང་ནས་ཐག་ཆོད་ཀྱི། །གདེངས་ཚད་དེ་འདྲར་དམྱལ་བ་ག་ནས་འོངས། །ད་དུང་སྨོན་ལམ་འདི་འདྲ་རྒྱུན་དུ་མཛོད། །ད་ནི་ནམ་མཁར་རྡོ་རྗེའི་གར་མཛོད་ཅིག །ཟེར་བས་བདག་ཉིད་ ༢༢༠༽བར་སྣང་ཁམས་སུ་འཕགས། །ཆོས་ཚིག་ཤུགས་འབྱུང་མང་པོ་གླུ་ལེན་ཅིང༌། པད་སྐོར་གར་སྟབས་མང་པོ་བྱས་པའི་འཕྲོར། །གཉིས་ལས་སད་ཅིང་ལུས་པོ་འདར་ཡོད་འདུག །འདི་འདྲའི་རྨི་ལམ་སྔོན་ཆད་རྨིས་མ་མྱོང༌། །མཚར་སྙམ་ཡི་གེའི་ལམ་དུ་བཀོད་ལགས་ཀྱང༌། །དུས་མཐའི་འགྲོ་བ་ལོག་ལྟ་ཆེ་བའི་མཐུས། །གཞན་ཡིད་ཁོང་ཁྲོའི་ཟུག་རྔུ་མི་ལྡང་ཤོག །ཚུལ་འདི་དད་ལྡན་འགའ་ལ་ཕན་སྲིད་ན། །ཐམས་ཅད་བློ་ཁ་ཆོས་ལ་འགྱུར་བ་ཤོག ༈ ཡང་ནི་ལྕགས་མོ་གླང་ལོ་ཆོ་འཕྲུལ་ཟླའི། །དུས་ཆེན་ཚེས་པ་བཅོ་ལྔའི་ནུབ་མོ་རུ། །དགེ་སློང་སྒྲིགས་མ་ཐམས་ཅད་ཚར་བའི་མཐར། །སླར་ཡང་ལྟོས་བཅས་འབྲེལ་ཐོགས་མཁའ་ཁྱབ་ཀྱི། །འགྲོ་ལ་དམིགས་ནས་སྒྱུ་ལུས་མཆོད་སྦྱིན་དང༌། །བཟང་སྤྱོད་ལ་སོགས་སྨོན་ལམ་མང་པོ་བཏོན། །ནམ་ཕྱེད་མ་སོང་བར་དུ་བསྡད་པའི་རྗེས། གཉིད་དུ་ཉལ་བའི་རྨི་ལམ་ཉིང་འཁྲུལ་ལ། སྔར་ནས་ ༢༢༡༽ཤི་བའི་རང་གི་གྲྭ་རྒན་རིགས། །གྲོང་ཆོག་མང་དུ་འགྲོ་མཁན་འགའ་རེ་དང༌། །གཉེར་བྱེད་ལས་འཛིན་ཤི་བ་འགའ་དང་བཅས། །ལྷན་ཅིག་མི་དགའ་བཞིན་དུ་དབང་མེད་ཕྱིན། །ཤ་ཟན་རོང་ཡིན་ཟེར་བའི་ལུང་པ་ནི། །རི་རོང་དོག་ཅིང་མུན་ནག་འཐིབ་པ་ཡི། །ནུབ་ནས་ཤར་དུ་ཁ་ལྟས་རི་གཟར་པོ། །རྩི་ཤིང་ལ་སོགས་གང་ཡང་མ་སྐྱེས་ཤིང༌། །ས་ཀུན་ཁྲག་གི་འདམ་རྫབ་ནང་ཁྲོལ་དང༌། །རྒྱུ་ལྟོ་ལྗན་ལྗིན་མང་པོས་ཁྱབ་པའི་ངོགས། །གཟར་མོའི་ཐུར་དེར་འདྲེད་རུང་འགྱེལ་རུང་ཕྱིན། །རྭ་གཞོབ་འདྲ་བའི་དྲི་མ་ངན་པ་འཚུབས། །རྐང་པ་ཐམས་ཅད་ཁྲག་གིས་འདམ་གྱིས་གཡོགས། ༢༢༢༽ །ཁང་ཧྲུལ་བུག་པའི་ནང་དུ་འཇུལ་ཏེ་སོང༌། །བདག་ཀྱང་ལམ་ངན་འདི་འདྲར་འགྲོ་བ་ལས། །ཁོང་ཚོའི་ཕྱི་ལ་བྱིན་པ་དྲག་གམ་སྙམ། །བརྫངས་པས་ཁོང་ཚོ་འཛུལ་སར་ང་མ་ཤོང། །ཡང་ནི་སྔར་གྱི་ཐུར་ལམ་འཕྲོ་དེར་ཕྱིན། །མར་མར་སླེབ་བཞིན་མུན་པ་ནག་ཏུ་སོང༌། །སྡོང་སྐས་གཟར་མོ་ཐོག་སོ་གསུམ་ཚད་ཙམ། །དེ་མགོར་སླེབ་ཚེ་ཤིན་ཏུ་ཁ་ནག་པས། །མ་མཐོང་སྦར་འཇུས་བྱས་ནས་བབས་པའི་དུས། །རང་གི་མགོ་འདི་རབ་ཏུ་ལྗིང་བ་འདྲས། །ཡོར་མ་ཐུབ་པར་མགོ་མཇུག་ལོག་ལ་ཁད། །སྐས་ལ་དམ་དུ་བཟུང་ཞིང་བབས་པའི་ཐིལ། །ཁང་ཁོག་ཆེན་པོ་ལུད་དང་འདམ་རྫབ་ཅན། །དྲི་མི་ཞིམ་ཞིང་ཁང་ནག་གཅིག་ཏུ་སླེབ། །ཁྱི་གུའམ་ནི་ཨ་ལི་ཕྲུ་གུའི་སྐད། །འདྲ་ལ་མི་སྙན་ཙི་དིར་མང་པོ་སྦིར། །དེ་ཡི་བར་ཁང་ཞིག་ཏུ་ཕྱིན་པ་ཡིས། །ངོ་མི་ཤེས་པའི་གཅན་གཟན་ཕྲུ་གུ་ནི། །མང་པོས་བདག་ ༢༢༣༽ལ་རུབ་ནས་འབྲད་དུ་བྱུང༌། །བེར་ཀ་གཡུག་ཅིང་རྒོལ་ཡང་མ་ཐུབ་པར། །བདག་གི་ལག་པ་གཡོན་ལ་གཅིག་གིས་རྨུགས། །བེར་ཀ་རྒྱབ་པས་དེ་ནི་ས་ལ་ལྷུང༌། །ལྕི་བའི་ཕུང་པོ་གཅིག་ཏུ་གྱུར་ནས་སྣང༌། །ལག་པའི་མཐིལ་གྱི་རྨ་ལ་བལྟས་ཙ་ན། །ལྕགས་ཀྱི་སྡེར་མོ་ཁ་གསུམ་རྣོན་པོ་ཞིག །ཟུགས་ནས་གདའ་བས་ནན་གྱིས་བཀོག་ནས་དེར། །མི་སྐྱ་གཉིས་འདུག་གཅིག་གི་པང་པར་བསྐྱུར། ༈ ཁྱེད་ཀྱི་ཁྱིའམ་གཅན་གཟན་འདི་རྣམས་ཀྱིས། །ང་ལ་འདི་ལྟར་རྨུགས་བྱུང་སྨྲས་པའི་ཚེ། །མི་སྐྱ་དེ་ཡིས་བདག་ལ་འདི་སྐད་ཟེར། ༈ འདི་ནི་ཤ་ཟན་སྲོག་གི་སྤུ་གྲི་ཡིན། །ཁྱེད་ཅག་བན་བཙུན་རྣམས་ལ་ལྷག་པར་སྡང༌། །ཁྱེད་ལ་ལག་པར་ཟུག་པ་གཡས་ཆུང་ཡིན། །གཞན་ནི་གླད་རྒྱའམ་སྙིང་རྩ་ནག་པོ་ལ། །ཟུག་ནས་ཐོན་པའི་གོ་སྐབས་མེད་པས་ན། །དེ་འདྲར་མ་ཟུག་ཁ་རྗེ་ ༢༢༤༽ཁྱེད་རེ་ཆེ། ༈ །དེ་སྐད་ཟེར་འཕྲོར་སྟེང་ཐོག་ཞིག་ཏུ་སླེབ། །རང་གི་ཉེ་གནས་བསྟན་འཛིན་དར་རྒྱས་ཀྱིས། །ཚོད་མ་འདྲ་བ་དེ་ལས་སེར་ཙམ་པ། །ཐུར་མ་གང་ཞིག་བདག་གི་ལག་ཏུ་བླུག

༢༢༧༽།བདག་ནི་རྨི་ལམ་ཡིན་པར་དྲན་མ་ཐུབ། །འཇིགས་ཀྱང་འཇིགས་ལ་སྐྱོ་ཡང་རབ་ཏུ་སྐྱོ། །མཆི་མ་འཁྲུག་བཞིན་ལན་དུ་འདི་སྨྲས་སོ། ༈ བདེན་ནོ་བདེན་ནོ་ཅི་གསུང་ཤིན་ཏུ་བདེན། །ངེད་ཅག་དུས་ངན་སྙིགས་མའི་བཙུན་གཟུགས་རྣམས། །མི་སྐྱ་ལས་ཀྱང་ལོད་པའི་སྤྱོད་ཡུལ་ལ། །ཁྲིམས་ཀྱི་ཆད་པས་གཅོད་པ་ཐུགས་བདེན་མོད། །བདག་གི་ངོས་ནས་རྒྱུ་མཚན་ཅི་ཞུ་ན། །བྱིས་པའི་དུས་ནས་བག་ཆགས་བཟང་བའི་རྒྱུས། །གཞན་ལྟར་སྡིག་པའི་དྲི་མས་གོས་མ་མྱོང༌། །ཁྱད་པར་ཤ་ཁྲག་དམར་ལ་ཞེན་པ་ལོག །སྡིག་ཟས་ཁོང་དུ་སོང་བ་ཉུང་ཤོས་ཡིན། །དཀར་དང་འབྲུ་འབྲས་གོ་མ་ལ་ཉུང་དང༌། །ཤ་མོ་ཚོད་མ་སྐྱོ་ཚའི་འཚོ་བ་ཡིས། །མི་ཚེ་འདི་ཙམ་བར་དུ་བསྐྱལ་པ་ལས། །ཤ་ཆང་ཟས་ལ་བསྟེན་པའི་མི་གཅིག་མིན། །བཟང་ཇ་རིན་ཐང་ཅན་རྣམས་འཐུང་མང་བའི། །དཀོར་སྒྲིབ་ ༢༢༨༽ཉེས་པ་ཆེ་བར་ཡོད་ཤས་ཆེ། །གཞན་ནི་ཉེས་ལྟུང་ཕོག་པའི་སྐྱོན་ཅན་ཞིག །ཡོང་བའི་རྒྱུ་རྐྱེན་བདག་ལ་ཡོད་སྙམ་མེད། །བསམ་པ་རྙེད་ཅིང་གོ་རྟོགས་འཇགས་ཚུན་ནས། །བཟའ་བཏུང་གདན་གོས་ཉེར་སྤྱད་མཐའ་དག་ལ། །འདུ་ཤེས་གསུམ་སོགས་སྐྱེད་རྫོགས་གང་རུང་གི །གནད་ཀྱིས་ཐ་མལ་ལོངས་སུ་སྤྱོད་མ་མྱོང༌། །ཐོས་བསམ་སྒོམ་པའི་གདེང་ཚད་མེད་ནའང༌། །དྲིན་ཅན་བླ་མ་བཅོ་བརྒྱད་ཞབས་ལ་ཟིན། །ཀུན་གྱིས་དགོངས་པའི་རྩེ་གཏད་ཐུགས་རྗེས་བསྐྱངས། །དམ་ཚིག་ཉམས་ཆག་དྲི་མས་གོས་སྙམ་མེད། །ན་སོ་ཆེར་སོང་དམ་བཅའ་མང་དུ་གཏོང༌། །དཀའ་སྤྱད་སྙིང་རུས་དགེ་སྦྱོར་གོང་དུ་སྤར། །འཆི་དུས་མི་འགྱོད་ཙམ་ཞིག་ཨེ་ཡོང་བསམ། །ཁྱེད་ཀྱི་ཁྲིམས་བཅད་མཛད་ནའང་གང་འོས་མཛོད། །ཁྱད་པར་ཤ་ཆང་བག་མེད་སྤྱོད་པ་ནི། །རང་ཡང་མ་བྱས་གཞན་ལའང་ཞེན་པ་ཆད། །མེ་ལོ་བྱ་ལོ་ས་ ༢༢༩༽གའི་དུས་ཆེན་ ནས། །བཟུང་སྟེ་མི་ཚེ་འདི་ལ་གར་སྡོད་ཀྱང༌། །ཕབས་ཀྱིས་བཏབ་པའི་རིགས་གྱུར་དྲི་ཙམ་ཡང༌། །ཕྱོགས་སུ་མི་ཉེ་ཐག་ཆོད་དམ་བཅའ་ནི། །བདག་ཅག་དཔོན་སློབ་གསུམ་ཀས་ལྷན་ཅིག་བྱས། །ཚེས་བཅུའི་མཆོད་པ་བདུད་རྩི་སྨན་རཀ་ཀྱང༌། །གཙང་མའི་ཆུ་ལ་ཀ་ར་བུ་རམ་སྦྱངས། །རྒྱུད་ནས་གསུངས་
